\subsection{A ordem dos nĂşmeros naturais}

Nesta seĂ§ĂŁo, estudaremos algumas propriedades bĂĄsicas acerca da ordem dos nĂşmeros naturais.

Das definiĂ§Ăľes de $1$ atĂŠ $9$, intuitivamente fica aparente que, para todo nĂşmero natural $n$, deverĂĄ valer $n = \{m \in \omega: m<n\}$.
Tal igualdade de fato serĂĄ verdadeira, como veremos nas prĂłximas pĂĄginas.
Por isso, parece natural definir que a ordem dos nĂşmeros naturais ĂŠ dada pela relaĂ§ĂŁo de pertencimento.

\begin{definition}
    Sejam $m, n \in \omega$.
    Dizemos que $m$ ĂŠ menor que $n$, e escrevemos $m < n$, se, e somente se $m \in n$.

    Define-se que $m\leq n$ se, e somente se $m < n$ ou $m = n$.
\end{definition}

\begin{lemma}\label{lem:ordemZero}
    Para todo $n \in \omega$, $0\leq n$.
\end{lemma}
\begin{proof}
    A prova segue por induĂ§ĂŁo em $n$.
    Para $n=0$, $0\leq 0$ pois $0=0$.
    
    Suponha que $0\leq n$.
    EntĂŁo $0<n$ ou $0=n$.

    Se $0<n$, temos que $0\in n\subseteq n\cup\{n\}=S(n)$, e, portanto, $0<S(n)$.
    Se $0=n$, entĂŁo $0\in \{0\}=S(0)=S(n)$, e, portanto, $0<S(n)$.
    De qualquer forma, $0\leq S(n)$.
\end{proof}

\begin{lemma}\label{lem:ordem-sucessor}
    Para quaisquer $m, n \in \omega$, $m < S(n)$ se e somente se $m \leq n$.
\end{lemma}
\begin{proof}
    Temos:
    \begin{align*}
        m < S(n) &\leftrightarrow m \in S(n)\\
        &\leftrightarrow m \in n \cup \{n\}\\
        &\leftrightarrow m \in n \lor m = n\\
        &\leftrightarrow m < n \lor m = n\\
        &\leftrightarrow m \leq n.
    \end{align*}
\end{proof}
Veremos que a ordem dos nĂşmeros naturais ĂŠ uma ordem total.
Iniciaremos pela transitividade.
\begin{lemma}
    Para quaisquer $l, m, n \in \omega$, se $l < m$ e $m < n$, entĂŁo $l < n$.
\end{lemma}
\begin{proof}
    Provaremos por induĂ§ĂŁo em $n$ que para todo $n \in \omega$, para quaisquer $l, m \in \omega$, se $l < m$ e $m < n$, entĂŁo $l < n$.

    Para $n=0$, a hipĂłtese $m < n$ ĂŠ falsa, e, portanto, a tese ĂŠ verdadeira.
    Suponha que a tese seja verdadeira para um $n \in \omega$.
    Devemos mostrar que para quaisquer $l, m \in \omega$, se $l < m$ e $m < S(n)$, entĂŁo $l < S(n)$.
    
    Assim, sejam $l, m \in \omega$ tais que $l < m$ e $m < S(n)$.
    Pelo Lema~\ref{lem:ordem-sucessor}, $l < m$ e $m \leq n$.

    Como $m\leq n$, temos que $m<n$ ou $m=n$.
    Se $m<n$, segue da hipĂłtese de induĂ§ĂŁo que $l < n$.
    Se $m=n$, entĂŁo $l < n$ pois $l < m$.
    Em qualquer caso, portanto, $l<n$.
    Assim, $l \in n \subseteq n \cup \{n\}=S(n)$, e, portanto, $l < S(n)$.
\end{proof}

Agora provaremos a irreflexibilidade da ordem dos nĂşmeros naturais.
\begin{lemma}
    Para todo $n \in \omega$, $n \not< n$.
\end{lemma}
\begin{proof}
    A prova segue por induĂ§ĂŁo em $n$.
    Para $n=0$, $0\not< 0$ pois $0\notin 0=\emptyset$.
    Suponha que $n \not< n$. Devemos ver que $S(n) \not< S(n)$.
    Suponha, para fins de contradiĂ§ĂŁo, que $S(n) < S(n)$.
    Pelo Lema~\ref{lem:ordem-sucessor}, $S(n) \leq n$.
    Ademais, $n< S(n)$.
    
    Se $S(n)<n$, segue da transitividade que $n<n$, o que nos dĂĄ um absurdo.
    JĂĄ se $S(n)=n$, entĂŁo substituindo em $n< S(n)$, temos novamente o absurdo $n<n$.
    Logo, $S(n)\not< S(n)$.
\end{proof}

Resta demonstrar a totalidade da ordem dos nĂşmeros naturais.
Para tanto, precisaremos de um lema auxiliar dual ao Lema~\ref{lem:ordem-sucessor}.

\begin{lemma}\label{lem:ordem-menor-que-sucessor}
    Para quaisquer $n, m \in \omega$, $S(n) \leq m$ se e somente se $n < m$.
\end{lemma}
\begin{proof}
    Se $S(n)\leq m$, entĂŁo $n<m$, pois $n<S(n)\leq m$ e vale a transitividade.

    Para a recĂ­proca, utilizaremos induĂ§ĂŁo em $m$ para mostrar que para todo $m \in \omega$, para quaisquer $n \in \omega$, se $n < m$, entĂŁo $S(n) \leq m$.

    Para $m=0$, a hipĂłtese $n < m$ ĂŠ falsa, e, portanto, a tese ĂŠ verdadeira.
    
    Suponha que a tese seja verdadeira para um $m \in \omega$.
    Seja $n \in \omega$ tal que $n < S(m)$.
    Assim, pelo Lema~\ref{lem:ordem-sucessor}, $n \leq m$.
    Se $n=m$, entĂŁo $S(n)=S(m)\leq S(m)$.
    Se $n<m$, entĂŁo, pela hipĂłtese de induĂ§ĂŁo, $S(n) \leq m<S(m)$.
    De qualquer forma, $S(n) \leq S(m)$.
\end{proof}

Agora estamos prontos para provar a totalidade da ordem dos nĂşmeros naturais.
\begin{proposition}
    $(\omega, <)$ ĂŠ uma ordem total estrita.
\end{proposition}
\begin{proof}
    JĂĄ vimos que $(\omega, <)$ ĂŠ uma ordem parcial estrita.
    Por induĂ§ĂŁo em $n$, veremos que para todo $n \in \omega$, para todo $m \in \omega$, $m < n$ ou $m = n$ ou $n < m$.

    Para $n=0$, a tese segue do Lema~\ref{lem:ordemZero}.

    Suponha que a tese seja verdadeira para um $n \in \omega$.
    Seja $m \in \omega$.
    Devemos ver que $m < S(n)$ ou $m = S(n)$ ou $S(n) < m$.

    Pela hipĂłtese de induĂ§ĂŁo, sabemos que $m<n$ ou $m = n$ ou $n < m$.

    Se $m\leq n$, temos que $m\leq n<S(n)$, logo, por transitividade, $m<S(n)$.
    JĂĄ se $n<m$, segue do Lema~\ref{lem:ordem-menor-que-sucessor} que $S(n)\le m$.
    Logo, $S(n)=m$ ou $S(n)<m$, o que conclui a prova.
\end{proof}

Nosso prĂłximo passo ĂŠ provar que a ordem dos nĂşmeros naturais ĂŠ uma boa ordem.
Primeiro provaremos que o conjunto dos nĂşmeros naturais satisfaz o esquema da induĂ§ĂŁo forte.

\begin{proposition}\label{prop:inducao-fort-set}
    Seja $A\subseteq \omega$.
    Se $\forall n \in \omega\, (\forall m < n\, (m \in A) \rightarrow n \in A)$, entĂŁo $A = \omega$.
\end{proposition}
\begin{proof}
    Ă suficiente provar que $\omega \subseteq A$.

    Seja $B=\{n \in \omega : \forall m < n\, (m \in A)\}$.
    Veremos que $B$ ĂŠ indutivo.
    Por vacuidade, $0\in B$.
    
    Suponha que $n \in B$.
    Vejamos que $S(n) \in B$.
    Seja $k < S(n)$.
    Devemos ver que $k \in A$.
    Pelo Lema~\ref{lem:ordem-sucessor}, $k \leq n$.
    Se $k<n$, entĂŁo $k \in A$ pois $n \in B$.
    Se $k=n$, temos que $\forall m < n\, (m \in A)\rightarrow n \in A$, e tambĂŠm que $\forall m < n\, (m \in A)$, pois $n \in B$.
    Assim, por modus ponens, $n \in A$.
    Isso conclui que $S(n) \in B$.
    Assim, $B$ ĂŠ indutivo.

    Logo, $\omega \subseteq B$.
    Seja $n \in \omega$.
    Temos que $S(n) \in B$ e que $n < S(n)$.
    Assim, $n \in A$.
\end{proof}

Disso, decorre o esquema da induĂ§ĂŁo forte.
\begin{proposition}
    Considere uma fĂłrmula $\phi(x, t_1, \dots, t_n)$.

    O seguinte ĂŠ um teorema:
    \[
    \forall t_1, \dots, t_n \Bigl(
        \bigl(
            \forall n \in \omega\, (\forall m < n\, \phi(m, t_1, \dots, t_n) \rightarrow \phi(n, t_1, \dots, t_n))
        \bigr)
        \rightarrow
        \forall n \in \omega\, \phi(n, t_1, \dots, t_n)
    \Bigr).
    \]
\end{proposition}
\begin{proof}
    Sejam dados $t_1, \dots, t_n$.
    Considere o conjunto $A = \{n \in \omega : \phi(n, t_1, \dots, t_n)\}$.
    Da ProposiĂ§ĂŁo~\ref{prop:inducao-fort-set} e da hipĂłtese, segue imediatamente que $A=\omega$.
\end{proof}

\begin{corollary}
    $(\omega, <)$ ĂŠ uma boa ordem.
\end{corollary}
\begin{proof}
    JĂĄ vimos que $(\omega, <)$ ĂŠ uma ordem total estrita.
    Resta mostrar que todo subconjunto nĂŁo vazio de $\omega$ tem um elemento mĂ­nimo.

    Seja $A\subseteq \omega$ sem elemento mĂ­nimo.
    Veremos que $A$ ĂŠ vazio.
    Para tanto, basta mostrar que $B=\omega\setminus A$ coincide com $\omega$.
    Pela ProposiĂ§ĂŁo~\ref{prop:inducao-fort-set}, ĂŠ suficiente mostrar que $\forall n \in \omega\, (\forall m < n\, (m \in B) \rightarrow n \in B)$.

    Ora, seja $n \in \omega$ tal que $\forall m < n\, (m \in B)$.
    Suponha, para fins de contradiĂ§ĂŁo, que $n \notin B$, ou seja, $n \in A$.
    Como $A$ nĂŁo possui mĂ­nimo, existe $m < n$ tal que $m \in A$.
    PorĂŠm, $m \in B$, o que ĂŠ uma contradiĂ§ĂŁo.
    Assim, $n \in B$.
\end{proof}
