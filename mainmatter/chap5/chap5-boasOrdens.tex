\section{Boas ordens}
Nesta seĂ§ĂŁo, nos aprofundaremos um pouco mais na noĂ§ĂŁo de boa ordem.

\begin{definition}
Seja $(A, <)$ uma boa ordem.
Dizemos que $S\subseteq A$ ĂŠ um \emph{segmento inicial} de $A$ se, para todo $x\in S$ e todo $y\in A$, se $y<x$, entĂŁo $y\in S$.
Dizemos que um segmento inicial $S$ de $A$ ĂŠ \emph{prĂłprio} se $S\neq A$.
\end{definition}

\begin{lemma}
    Seja $A$ uma boa ordem e $\phi: A\to A$ uma funĂ§ĂŁo crescente.
    EntĂŁo, para todo $a\in A$, $a\leq \phi(a)$.
\end{lemma}
\begin{proof}
    Se nĂŁo, seja $a=\min\{x\in A:\phi(x)<x\}$.
    Como $\phi$ ĂŠ crescente e $\phi(a)<a$, temos
    \[
        \phi(\phi(a))<\phi(a).
    \]
    Mas $\phi(a)<a$, de modo que, pela minimalidade de $a$, nĂŁo pode valer
    \[
        \phi(\phi(a))<\phi(a).
    \]
    ContradiĂ§ĂŁo.
\end{proof}

\begin{lemma}
    Seja $S$ um segmento inicial de uma boa ordem $A$ e $f:A\to S$ um isomorfismo.
    EntĂŁo $S=A$ e $f$ ĂŠ a funĂ§ĂŁo identidade de $A$.
\end{lemma}
\begin{proof}
    Pelo lema anterior, para todo $a\in A$, $a\leq f(a)$.
    Como $f(a)\in S$ e $S$ ĂŠ um segmento inicial de $A$, segue que $a\in S$.
    Assim, $S=A$.

    Pelo lema anterior novamente aplicado Ă  $f^{-1}: A\to A$, para todo $a\in A$, $a\leq f^{-1}(a)$.
    Aplicando $f$ em ambos os lados, temos que $f(a)\leq a$.
    Assim, $a=f(a)$ para todo $a\in A$.
\end{proof}

\begin{corollary}
    Sejam $A$ e $B$ boas ordens.
    Se $A$ ĂŠ isomorfo Ă  $B$, entĂŁo existe um Ăşnico isomorfismo de $A$ em $B$.
\end{corollary}
\begin{proof}
    Sejam $f, g: A\to B$ isomorfismos.
    EntĂŁo $g^{-1}\circ f: A\to A$ ĂŠ um isomorfismo.
    Pelo lema anterior, $g^{-1}\circ f$ ĂŠ a funĂ§ĂŁo identidade de $A$, e, assim, $f=g$.
\end{proof}

\begin{definition}
    Seja $A$ uma boa ordem e $a\in A$.
    Definimos $A[a] = \{x\in A : x<a\}$.
\end{definition}
\begin{lemma}
    Seja $A$ uma boa ordem e $a\in A$.
    EntĂŁo $A[a]$ ĂŠ um segmento inicial prĂłprio de $A$.
    Reciprocamente, se $S$ ĂŠ um segmento inicial prĂłprio de $A$, entĂŁo existe um Ăşnico $a\in A$ tal que $S=A[a]$.
\end{lemma}
\begin{proof}
    Seja $a\in A$.
    Para todo $x\in A[a]$ e todo $y\in A$, se $y<x$, entĂŁo $y\in A[a]$, logo, $A[a]$ ĂŠ um segmento inicial de $A$.
    Ele ĂŠ prĂłprio, pois $a\in A$ e $a\notin A[a]$.

    Seja $S$ um segmento inicial prĂłprio de $A$.
    Seja $a=\min\{x\in A : x\notin S\}$.
    Temos que $A[a]\subseteq S$ pela minimalidade de $a$.
    AlĂŠm disso, $S\subseteq A[a]$, pois, se $x\in S$ e $x\notin A[a]$, entĂŁo $a<x$, o que implica que $a\in S$, um absurdo.
\end{proof}

\begin{lemma}
    Sejam $A$ e $B$ boas ordens e $f:A\to B$ um isomorfismo.
    EntĂŁo, para todo $a\in A$, $f|A[a]: A[a]\to B[f(a)]$ ĂŠ um isomorfismo.
\end{lemma}
\begin{proof}
    Ă claro que $f|A[a]: A[a]\to B$ ĂŠ uma funĂ§ĂŁo crescente, e, portanto, injetora.

    Se $x\in A[a]$, entĂŁo $x<a$, e, como $f$ ĂŠ crescente, $f(x)<f(a)$, ou seja, $f(x)\in B[f(a)]$.
    Assim, $f|A[a]: A[a]\to B[f(a)]$ ĂŠ uma funĂ§ĂŁo crescente.

    Se $y \in B[f(a)]$, entĂŁo $y<f(a)$, e, como $f$ ĂŠ sobrejetora, existe um Ăşnico $x\in A$ tal que $f(x)=y<f(a)$.
    Como $f$ ĂŠ um isomorfismo, temos que $x<a$, ou seja, $x\in A[a]$.
    Assim, $f|A[a]: A[a]\to B[f(a)]$ ĂŠ uma funĂ§ĂŁo bijetora, e, portanto, um isomorfismo.
\end{proof}

\begin{lemma}\label{lem:segmentos-iniciais}
    Seja $A$ uma boa ordem e $b<a$ elementos de $A$.
    EntĂŁo $(A[a])[b]=A[b]$.
\end{lemma}
\begin{proof}
    ExercĂ­cio.
\end{proof}

\begin{theorem}
    Sejam $A$ e $B$ boas ordens.
    EntĂŁo uma, e somente uma das seguintes alternativas ĂŠ verdadeira:
\end{theorem}
\begin{enumerate}[label=(\alph*)]
    \item $A$ ĂŠ isomorfo Ă  $B$;
    \item $A$ ĂŠ isomorfo a um segmento inicial prĂłprio de $B$;
    \item $B$ ĂŠ isomorfo a um segmento inicial prĂłprio de $A$.
\end{enumerate} 
Em qualquer caso, o isomorfismo ĂŠ Ăşnico.
\begin{proof}
    A unicidade do isomorfismo decorre do corolĂĄrio anterior.
    Iniciaremos mostrando que somente uma alternativa ĂŠ possĂ­vel.
    Se valem (a) e (b), compondo os isomorfismos, temos que $A$ ĂŠ isomorfo a um segmento inicial prĂłprio de $A$, o que ĂŠ impossĂ­vel pelo segundo lema desta seĂ§ĂŁo.
    Analogamente, (a) e (c) nĂŁo podem valer simultaneamente.

    Se valem (b) e (c), entĂŁo $A$ ĂŠ isomorfo a um segmento inicial prĂłprio de $B$, digamos, $B[b]$, e $B$ ĂŠ isomorfo a um segmento inicial prĂłprio de $A$, digamos, $A[a]$.
    Seja $f:B\to A[a]$ um isomorfismo.
    EntĂŁo $f|B[b]: B[b]\to (A[a])[f(b)]=A[f(b)]$ ĂŠ um isomorfismo, e, assim, $A$ ĂŠ isomorfo a um segmento inicial prĂłprio de $A$, o que ĂŠ impossĂ­vel.

    Agora provaremos que um dos itens vale.
    Seja $f=\{(a, b)\in A\times B : A[a]\text{ ĂŠ isomorfo a }B[b]\}$.

    $\dom f$ ĂŠ um segmento inicial de $A$: se $a\in \dom f$ e $x<a$, entĂŁo existe um isomorfismo $g: A[a]\to B[b]$ para algum $b\in B$.
    EntĂŁo $g|A[x]: A[x]\to B[g(x)]$ ĂŠ um isomorfismo, e, assim, $x\in \dom f$.

    Analogamente, $\ran f$ ĂŠ um segmento inicial de $B$.

    $f$ ĂŠ uma funĂ§ĂŁo: se $(a, b), (a, b')\in f$, entĂŁo $B[b]$ e $B[b']$ sĂŁo isomorfos.
    Como um deles ĂŠ um segmento inicial do outro, eles sĂŁo iguais, e, portanto, $b=b'$.

    $f$ ĂŠ crescente: se $a<a'$ estĂŁo em $\dom f$, entĂŁo existe um isomorfismo $g: A[a']\to B[f(a')]$.
    Temos que $g|A[a]: A[a]\to B[g(a)]$ ĂŠ um isomorfismo, e, assim, $f(a)=g(a)<g(a')=f(a')$.

    Se $A=\dom f$ ou $B=\ran f$, entĂŁo uma das trĂŞs alternativas ĂŠ verdadeira.
    
    JĂĄ se $A\neq \dom f$ e $B\neq \ran f$, $\dom f=A[a]$ e $\ran f=B[b]$ para algum $a\in A$ e $b\in B$.
    Pelo exposto acima, $f: A[a]\to B[b]$ ĂŠ um isomorfismo, e, assim, $(a, b)\in f$, logo, $a \in \dom f=A[a]$, um absurdo.
\end{proof}

\subsection{ExercĂ­cios}
\begin{exercise}
    Prove o Lema~\ref{lem:segmentos-iniciais}.
\end{exercise}
\begin{exercise}
    Prove que se $(A, <_A)$ e $(B, <_B)$ sĂŁo boas ordens, entĂŁo $A\times B$ ĂŠ uma boa ordem com a ordem lexicogrĂĄfica, ou seja, com a ordem dada por $(a, b)<(a', b')$ se $a<_A a'$, ou ($a=a'$ e $b<_B b'$).
\end{exercise}
\begin{exercise}
    De forma mais geral, seja $(I, <_I)$ uma boa ordem e, para cada $i\in I$, seja $(A_i, <_{A_i})$ uma boa ordem.
    Prove que $\bigcup_{i\in I} \{i\}\times A_i$ ĂŠ uma boa ordem com a ordem lexicogrĂĄfica, ou seja, com a ordem dada por $f<g$ se, e somente se $f\neq g$ e $f(i)<_{A_i} g(i)$, onde $i=\min\{j\in I : f(j)\neq g(j)\}$.
\end{exercise}
\begin{exercise}
Em particular, mostre que se $(A_0, <_{A_0})$ e $(A_1, <_{A_1})$ sĂŁo boas ordens, entĂŁo $\{0\}\times A_0\cup \{1\}\times A_1$ ĂŠ uma boa ordem com a ordem lexicogrĂĄfica, ou seja, com a ordem dada por $(i, a)<(j, b)$ se $i<j$, ou ($i=j$ e $a<_{A_i} b$).

Mostre ainda que se $A_0$ e $A_1$ sĂŁo conjuntos disjuntos, e $B=A_0\cup A_1$, e $<_B$ ĂŠ a ordem dada por $a<_B b$ se $a, b\in A_i$ e $a<_{A_i} b$, ou $a\in A_0$ e $b\in A_1$, entĂŁo $(B, <_B)$ ĂŠ uma boa ordem isomorfa a $\{0\}\times A_0\cup \{1\}\times A_1$.
Mostre que $(A_0, <_{A_0})$ ĂŠ um segmento inicial de $(B, <_B)$, e que este ĂŠ um segmento inicial prĂłprio se, e somente se, $A_1\neq \emptyset$.
\end{exercise}
