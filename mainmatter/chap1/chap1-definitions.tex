\section{AdiĂ§ĂŁo de sĂ­mbolos por definiĂ§ĂŁo}\label{sec:adicaoSimbolosPorDefinicao}
\index{definiĂ§ĂŁo!extensĂŁo por definiĂ§ĂŁo}

A Teoria dos Conjuntos ĂŠ uma teoria cuja linguagem nĂŁo lĂłgica inicial consiste apenas no sĂ­mbolo de pertencimento ($\in$).
Conforme veremos, precisaremos rapidamente de sĂ­mbolos funcionais adicionais, como o sĂ­mbolo de constante $\emptyset$ (que representa o conjunto vazio), o operador binĂĄrio de uniĂŁo de dois conjuntos ($\cup$) e o operador unĂĄrio do conjunto das partes ($\mathcal{P}$).
Analogamente, tambĂŠm passaremos a utilizar sĂ­mbolos relacionais adicionais, como o binĂĄrio $\subseteq$.

Neste texto, tais sĂ­mbolos serĂŁo introduzidos \emph{por definiĂ§ĂŁo}.
Intuitivamente, isto significa que eles servem apenas como abreviaĂ§Ăľes: em vez de escrever repetidamente uma fĂłrmula longa, introduzimos um sĂ­mbolo novo cujo comportamento ĂŠ \emph{explicitamente} fixado por um axioma.
O ponto central desta seĂ§ĂŁo ĂŠ formalizar essa prĂĄtica e justificar que ela nĂŁo aumenta o poder de prova da teoria em relaĂ§ĂŁo Ă s sentenĂ§as da linguagem original.

\subsection{O quantificador \texorpdfstring{$\exists!$}{â!}}\label{subsec:quantificadorExistsUnique}
\index{quantificador!existĂŞncia Ăşnica}

Os quantificadores primitivos da lĂłgica de primeira ordem, neste texto, consistem apenas do quantificador universal ($\forall$) e do quantificador existencial ($\exists$).

Um quantificador amplamente usado em MatemĂĄtica ĂŠ o quantificador existencial Ăşnico, denotado por $\exists!$.
Intuitivamente, $\exists! x \varphi$ significa que existe exatamente um $x$ tal que $\varphi$ ĂŠ verdadeira.

Introduziremos este operador como uma abreviaĂ§ĂŁo, conforme explicitado na definiĂ§ĂŁo a seguir.

\begin{metadefinition}[Quantificador de existĂŞncia Ăşnica]\label{mdef:quantificadorExistenciaUnica}\index{existĂŞncia Ăşnica}
    Seja $\varphi$ uma fĂłrmula.
    A notaĂ§ĂŁo $\exists!x\,\varphi$ ĂŠ uma abreviaĂ§ĂŁo para a fĂłrmula
    \begin{equation}\label{eq:existsUnique}
        (\exists x\,\varphi)\ \wedge\ \forall x\,\forall y\bigl((\varphi \wedge \varphi[x/y])\rightarrow x=y\bigr),
    \end{equation}
        em que $y$ ĂŠ a primeira variĂĄvel distinta de $x$ que nĂŁo ocorre em $\varphi$.
\end{metadefinition}

Na MetadefiniĂ§ĂŁo~\ref{mdef:quantificadorExistenciaUnica}, a variĂĄvel $y$ ĂŠ tomada como uma variĂĄvel auxiliar.
NĂŁo seria razoĂĄvel que a escolha de tal variĂĄvel dependesse da letra escolhida para representĂĄ-la.
Isso ĂŠ formalizado pelo resultado a seguir.


\begin{metalemma}[IndependĂŞncia do auxiliar em $\exists!$]\label{mlem:independenciaAuxiliarExistsUnique}
    Seja $\varphi$ uma fĂłrmula.
    Se $y$ e $z$ sĂŁo variĂĄveis distintas de $x$, distintas entre si, e que nĂŁo ocorrem em $\varphi$, entĂŁo as duas fĂłrmulas obtidas ao expandir
    $\exists!x\,\varphi$ pela EquaĂ§ĂŁo~\eqref{eq:existsUnique} usando $y$ ou usando $z$ sĂŁo equivalentes em qualquer teoria de primeira ordem.
\end{metalemma}

\begin{proof}
    Seja $T$ uma teoria de primeira ordem.
    Sejam $y, z$ variĂĄveis que nĂŁo ocorrem em $\varphi$, distintas de $x$ e distintas entre si.
    Suponha que $T\vdash (\exists x\,\varphi)\ \wedge\ \forall x\,\forall y\bigl((\varphi \wedge \varphi[x/y])\rightarrow x=y\bigr)$.
    Veremos que $T\vdash (\exists x\,\varphi)\ \wedge\ \forall x\,\forall z\bigl((\varphi \wedge \varphi[x/z])\rightarrow x=z\bigr)$.
    A recĂ­proca ĂŠ anĂĄloga.

    A parte existencial $\exists x\,\varphi$ jĂĄ ĂŠ a mesma nas duas expansĂľes.
    Para a parte de unicidade, pela comutatividade dos quantificadores universais obtemos
    \[
        T\vdash \forall y\,\forall x\bigl((\varphi \wedge \varphi[x/y])\rightarrow x=y\bigr).
    \]
    Como $z$ nĂŁo ocorre em $\varphi$, podemos renomear o quantificador externo de $y$ para $z$.
    Usando o Metalema~\ref{mlem:renomeacaoVariavelUniversal} e a igualdade
    $\varphi[x/y][y/z]=\varphi[x/z]$, concluĂ­mos
    \[
        T\vdash \forall z\,\forall x\bigl((\varphi \wedge \varphi[x/z])\rightarrow x=z\bigr).
    \]
    Aplicando novamente a comutatividade de quantificadores universais, segue
    \[
        T\vdash \forall x\,\forall z\bigl((\varphi \wedge \varphi[x/z])\rightarrow x=z\bigr),
    \]
    como querĂ­amos.
\end{proof}

\subsection{Definindo definiĂ§Ăľes}\label{subsec:definindoDefinicoes}
Agora estamos prontos para falar formalmente sobre definiĂ§Ăľes.

\begin{metadefinition}[DefiniĂ§ĂŁo explĂ­cita]\label{mdef:definicaoExplicita}\index{definiĂ§ĂŁo!explĂ­cita}\index{sĂ­mbolo!nĂŁo lĂłgico!definido}
Seja $T$ uma teoria de primeira ordem cujo lĂŠxico ĂŠ $\mathcal L$, e seja $\mathcal L'$ um lĂŠxico que estende $\mathcal L$.

\begin{enumerate}[label=(\roman*)]
    \item \textbf{Predicados.}
    Seja $P\in\mathcal L'\setminus\mathcal L$ um sĂ­mbolo de predicado $n$-ĂĄrio.
    Uma \emph{$T$-definiĂ§ĂŁo} de $P$ ĂŠ uma sentenĂ§a da forma
    \[
        \forall x_1\ldots \forall x_n\bigl(P(x_1,\ldots,x_n)\leftrightarrow \varphi(x_1,\ldots,x_n)\bigr),
    \]
    em que $x_1,\ldots,x_n$ sĂŁo variĂĄveis duas a duas distintas e $\varphi$ ĂŠ uma $\mathcal L$-fĂłrmula cujas variĂĄveis livres estĂŁo entre $x_1,\ldots,x_n$.
    No caso $n=0$, uma $T$-definiĂ§ĂŁo de $P$ ĂŠ uma sentenĂ§a $P\leftrightarrow \varphi$,
    com $\varphi$ uma $\mathcal L$-sentenĂ§a.

    \item \textbf{FunĂ§Ăľes.}
    Seja $f\in\mathcal L'\setminus\mathcal L$ um sĂ­mbolo de funĂ§ĂŁo $n$-ĂĄrio, com $n\geq 1$.
    Uma \emph{$T$-definiĂ§ĂŁo} de $f$ ĂŠ uma sentenĂ§a da forma
    \[
        \forall x_1\ldots \forall x_n\,\forall y\bigl(f(x_1,\ldots,x_n)=y \leftrightarrow \varphi(x_1,\ldots,x_n,y)\bigr),
    \]
    em que $x_1,\ldots,x_n,y$ sĂŁo variĂĄveis duas a duas distintas e $\varphi$ ĂŠ uma $\mathcal L$-fĂłrmula cujas variĂĄveis livres estĂŁo entre $x_1,\ldots,x_n,y$, tal que
    \[
        T\vdash \forall x_1\ldots \forall x_n\,\exists!y\,\varphi(x_1,\ldots,x_n,y).
    \]
    (Intuitivamente, $T$ prova que $\varphi$ define \emph{unicamente} um valor $y$ para cada entrada $x_1,\ldots,x_n$.)

    \item \textbf{Constantes.}
    Seja $c\in\mathcal L'\setminus\mathcal L$ um sĂ­mbolo de constante.
    Uma \emph{$T$-definiĂ§ĂŁo} de $c$ ĂŠ uma sentenĂ§a da forma
    \[
        \forall y\bigl(c=y \leftrightarrow \varphi(y)\bigr),
    \]
    em que $\varphi$ ĂŠ uma $\mathcal L$-fĂłrmula cujas variĂĄveis livres estĂŁo entre $y$, tal que $T\vdash \exists!y\,\varphi(y)$.
\end{enumerate}

Em cada um dos casos acima, dizemos que $\varphi$ ĂŠ a \emph{fĂłrmula definidora} do sĂ­mbolo novo.
Quando for conveniente, escreveremos essa fĂłrmula como
\[
\Def_P(x_1,\ldots,x_n),\qquad \Def_f(x_1,\ldots,x_n,y),\qquad \Def_c(y),
\]
conforme o caso.

AlĂŠm disso, se $P$, $f$ ou $c$ sĂŁo sĂ­mbolos dos tipos correspondentes em $\mathcal L$, convencionamos:

\begin{itemize}
    \item $\Def_P(x_1,\ldots,x_n)$ ĂŠ a fĂłrmula $P(x_1,\ldots,x_n)$.
    No caso $n=0$, $\Def_P$ ĂŠ a fĂłrmula $P$.
    \item $\Def_f(x_1,\ldots,x_n,y)$ ĂŠ a fĂłrmula $f(x_1,\ldots,x_n)=y$.
    \item $\Def_c(y)$ ĂŠ a fĂłrmula $c=y$.
    \item Se $z$ ĂŠ uma variĂĄvel, entĂŁo $\Def_{z}(z, y)$ ĂŠ a fĂłrmula $y=z$.
\end{itemize}

Note que, nesses casos, $\Def_P$, $\Def_f$, $\Def_c$ e $\Def_{z}$ sĂŁo fĂłrmulas da linguagem de $\mathcal L$.
\end{metadefinition}

\begin{example}\label{exa:definicoesExplicitas}
Ainda que nĂŁo tenhamos apresentado a Teoria dos Conjuntos, vejamos alguns exemplos.
\begin{itemize}
    \item \textbf{Teoria dos conjuntos.}
    Uma definiĂ§ĂŁo natural de $x\subseteq y$ na linguagem de $\in$ ĂŠ:
    \[
        \forall x\,\forall y\Bigl(x\subseteq y\ \leftrightarrow\ \forall z\,(z\in x\rightarrow z\in y)\Bigr).
    \]
    Aqui $\subseteq$ ĂŠ um novo sĂ­mbolo de predicado binĂĄrio, definido por uma $\mathcal L$-fĂłrmula (em que $\mathcal L=\{\in\}$).
    Nesse caso, $\Def_{\subseteq}$ ĂŠ a fĂłrmula $\forall z\,(z\in x\rightarrow z\in y)$.

    \item \textbf{Teoria dos grupos.}
    Considere a teoria de grupos na linguagem \emph{sem} sĂ­mbolo de constante para a identidade,
    isto ĂŠ, apenas com o sĂ­mbolo de funĂ§ĂŁo binĂĄria $\cdot$.
    Uma definiĂ§ĂŁo explĂ­cita (quando possĂ­vel em uma teoria dada) para um sĂ­mbolo novo de constante $e$ ĂŠ:
    \[
        \forall y\Bigl(e=y\ \leftrightarrow\ \forall x\,\bigl(x\cdot y=x \wedge y\cdot x=x\bigr)\Bigr).
    \]
    A condiĂ§ĂŁo de aplicabilidade ĂŠ que a teoria em questĂŁo prove $\exists!y\,\forall x(x\cdot y=x\wedge y\cdot x=x)$.
    Em uma axiomatizaĂ§ĂŁo usual de grupos \emph{sem} constante, isso ĂŠ demonstrĂĄvel.
    Nesse caso, $\Def_e$ ĂŠ a fĂłrmula $\forall x\,\bigl(x\cdot y=x \wedge y\cdot x=x\bigr)$.
\end{itemize}
\end{example}

Agora podemos definir a noĂ§ĂŁo de extensĂŁo por definiĂ§ĂŁo, uma formalizaĂ§ĂŁo da prĂĄtica de introduzir definiĂ§Ăľes como abreviaĂ§Ăľes.

\begin{metadefinition}[ExtensĂŁo por definiĂ§ĂŁo]\label{mdef:extensaoPorDefinicao}\index{definiĂ§ĂŁo!extensĂŁo por definiĂ§ĂŁo}
    Seja $T$ uma teoria de primeira ordem cujo lĂŠxico ĂŠ $\mathcal L$.
    Dizemos que uma teoria $T'$ ĂŠ uma \emph{extensĂŁo por definiĂ§ĂŁo} de $T$ se:
    \begin{itemize}
        \item o lĂŠxico de $T'$ ĂŠ um lĂŠxico $\mathcal L'$ que estende $\mathcal L$;
        \item os axiomas de $T'$ consistem nos axiomas de $T$ e, para cada sĂ­mbolo novo de $\mathcal L'\setminus\mathcal L$,
        em uma (Ăşnica) definiĂ§ĂŁo explĂ­cita no sentido da MetadefiniĂ§ĂŁo~\ref{mdef:definicaoExplicita}.
    \end{itemize}
\end{metadefinition}

Essa noĂ§ĂŁo pode causar estranhamento: estamos formalizando a introduĂ§ĂŁo de definiĂ§Ăľes como a adiĂ§ĂŁo de novos axiomas e, portanto, alterando a teoria base.
Isso ĂŠ necessĂĄrio no formalismo adotado, pois ele nĂŁo captura diretamente a prĂĄtica informal de introduzir definiĂ§Ăľes como meras abreviaĂ§Ăľes.

A seguir, veremos que extensĂľes por definiĂ§ĂŁo nĂŁo aumentam nem o poder de prova sobre sentenĂ§as da linguagem original, nem o poder de expressĂŁo.
Tudo o que ĂŠ expressĂĄvel ou demonstrĂĄvel na linguagem estendida tambĂŠm ĂŠ expressĂĄvel ou demonstrĂĄvel na linguagem original, respectivamente.
Portanto, apesar de, formalmente, estarmos adicionando novos sĂ­mbolos e axiomas, a extensĂŁo por definiĂ§ĂŁo captura as propriedades essenciais da prĂĄtica de introduzir definiĂ§Ăľes como abreviaĂ§Ăľes.


\subsection{EliminaĂ§ĂŁo de sĂ­mbolos definidos: termos}\label{subsec:eliminacaoSimbolosDefinidosTermos}
\index{eliminaĂ§ĂŁo de sĂ­mbolos definidos}

A ideia central ĂŠ associar a cada termo $t$ da linguagem $\mathcal L'$ uma fĂłrmula da linguagem $\mathcal L$ que semanticamente descreve $t$.
Isto substitui termos com sĂ­mbolos definidos por uma descriĂ§ĂŁo na linguagem original.

\begin{metadefinition}[FĂłrmula do valor de um termo]\label{mdef:formulaDoValorDeUmTermo}\index{termo!fĂłrmula do valor}\index{eliminaĂ§ĂŁo de sĂ­mbolos definidos!termos}
    Suponha que $T'$ ĂŠ uma extensĂŁo por definiĂ§ĂŁo de $T$, com lĂŠxicos $\mathcal L'$ e $\mathcal L$ respectivamente.
    Seja $t$ um termo de $\mathcal L'$.
    Definimos por recursĂŁo uma $\mathcal L$-fĂłrmula $\delta_t$ cujas variĂĄveis livres estĂŁo entre as variĂĄveis de $t$ e mais uma variĂĄvel $y$, lida como
    ``$y$ ĂŠ o valor do termo $t$'', como segue.

    \begin{enumerate}[label=(\roman*)]
        \item Se $t$ ĂŠ um sĂ­mbolo de constante $c$, entĂŁo $\delta_t(y)$ ĂŠ a fĂłrmula $\Def_c(y)$.
        \item Se $t$ ĂŠ uma variĂĄvel $z$, entĂŁo $\delta_t(z,y)$ ĂŠ a fĂłrmula $y=z$, ou seja, $\Def_z(z,y)$.
        \item Se $t$ ĂŠ $f(t_1,\ldots,t_m)$, entĂŁo $\delta_t(x_1, \dots, x_n,y)$ ĂŠ:
        \[
            \exists y_1\ldots \exists y_m\Bigl(
            \bigwedge_{i=1}^m \delta_{t_i}(x_1, \dots, x_n,y_i)\ \wedge\ \Def_f(y_1,\ldots,y_m,y)\Bigr),
        \]
        em que $x_1, \dots, x_n$ ĂŠ uma lista que contĂŠm todas as variĂĄveis que ocorrem em $t$,
        e $y_1, \dots, y_m$ sĂŁo variĂĄveis auxiliares duas a duas distintas e distintas de $x_1,\ldots,x_n,y$.
    \end{enumerate}
\end{metadefinition}

\begin{metalemma}[EliminaĂ§ĂŁo de termos]\label{mlem:eliminacaoDeTermos}\index{eliminaĂ§ĂŁo de sĂ­mbolos definidos!termos}
    Seja $T'$ uma extensĂŁo por definiĂ§ĂŁo de $T$, com $\mathcal L\subseteq \mathcal L'$.
    Seja $t$ um termo de $\mathcal L'$ cujas variĂĄveis estejam entre $x_1,\ldots,x_n$,
    e seja $y$ uma variĂĄvel distinta de $x_1,\ldots,x_n$.
    Seja $\delta_t(x_1,\ldots,x_n,y)$ como na MetadefiniĂ§ĂŁo~\ref{mdef:formulaDoValorDeUmTermo}. EntĂŁo:
    \begin{enumerate}[label=(\arabic*)]
        \item $T'\vdash \forall x_1\ldots \forall x_n\,\forall y\bigl(t=y \leftrightarrow \delta_t(x_1,\ldots,x_n,y)\bigr)$.
        \item $T\vdash \forall x_1\ldots \forall x_n\,\exists!y\,\delta_t(x_1,\ldots,x_n,y)$.
    \end{enumerate}

    AlĂŠm disso, se $t$ ĂŠ um $\mathcal L$-termo, entĂŁo:
    \begin{enumerate}[label=(\arabic*)]
        \setcounter{enumi}{2}
        \item $T\vdash \forall x_1\ldots \forall x_n\,\forall y\bigl(t=y \leftrightarrow \delta_t(x_1,\ldots,x_n,y)\bigr)$.
    \end{enumerate}
\end{metalemma}

\begin{proof}
Provaremos (1), (2) e (3) simultaneamente por induĂ§ĂŁo sobre a complexidade do termo $t$.
Escreva $\bar x$ para $x_1,\ldots,x_n$.

\smallskip
\textbf{Base 1: $t$ ĂŠ uma constante $c$.}
 Pela MetadefiniĂ§ĂŁo~\ref{mdef:formulaDoValorDeUmTermo}, $\delta_t(y)$ ĂŠ $\Def_c(y)$.

\emph{(1).} Se $c\in\mathcal L'\setminus\mathcal L$, entĂŁo a sentenĂ§a definidora de $c$ pertence a $T'$ e tem a forma
$\forall y\,(c=y\leftrightarrow \Def_c(y))$, que ĂŠ exatamente (1).
Se $c\in\mathcal L$, entĂŁo por convenĂ§ĂŁo $\Def_c(y)$ ĂŠ $c=y$, logo
$\forall y\,(c=y\leftrightarrow \Def_c(y))$ ĂŠ uma instĂ˘ncia de tautologia.

\emph{(2).} Se $c$ ĂŠ novo, a prĂłpria hipĂłtese de ``introduĂ§ĂŁo por definiĂ§ĂŁo'' garante
$T\vdash \exists!y\,\Def_c(y)$.
Se $c\in\mathcal L$, entĂŁo $\Def_c(y)$ ĂŠ $c=y$; provamos $T\vdash \exists!y\,(c=y)$:
\begin{itemize}
    \item \emph{ExistĂŞncia.} Pela reflexividade da igualdade, $T\vdash c=c$.
    Aplicando FO15 Ă  fĂłrmula $\phi(y)\equiv (c=y)$ com o termo $t=c$, obtemos o axioma lĂłgico
    $(c=c)\rightarrow \exists y(c=y)$; por \emph{modus ponens}, $T\vdash \exists y(c=y)$.
    \item \emph{Unicidade.} Dos axiomas de igualdade, temos $c=y\rightarrow y=c$ (simetria) e
    $y=c\wedge c=z\rightarrow y=z$ (transitividade). Assim,
    $(c=y\wedge c=z)\rightarrow y=z$, e portanto a parte de unicidade de $\exists!y(c=y)$.
\end{itemize}

\emph{(3).} Se $t$ ĂŠ um $\mathcal L$-termo, entĂŁo necessariamente $c\in\mathcal L$ e, como acima,
$\forall y\,(c=y\leftrightarrow \Def_c(y))$ ĂŠ tautolĂłgica; logo vale em $T$.

\smallskip
\textbf{Base 2: $t$ ĂŠ uma variĂĄvel $z$.}
Pela MetadefiniĂ§ĂŁo~\ref{mdef:formulaDoValorDeUmTermo}, $\delta_t(z,y)$ ĂŠ $y=z$, isto ĂŠ, $\Def_{z}(z,y)$.

\emph{(1).} Queremos $T'\vdash \forall z\,\forall y\,(z=y\leftrightarrow y=z)$.
Pela simetria da igualdade, temos
$T'\vdash z=y\rightarrow y=z$ e tambĂŠm $T'\vdash y=z\rightarrow z=y$.
Por uma instĂ˘ncia da tautologia proposicional
\[
(P\rightarrow Q)\rightarrow\bigl((Q\rightarrow P)\rightarrow (P\leftrightarrow Q)\bigr),
\]
obtemos $T'\vdash z=y\leftrightarrow y=z$, e tomando fecho universal em $z,y$ concluĂ­mos (1).

\emph{(2).} Provamos $T\vdash \forall z\,\exists!y\,(y=z)$.
A existĂŞncia segue de $T\vdash z=z$ e de FO15 aplicado a $\phi(y)\equiv (y=z)$ com o termo $t=z$.
A unicidade segue de $(y=z\wedge w=z)\rightarrow y=w$, obtida por simetria e transitividade da igualdade.
Logo vale (2) no caso variĂĄvel.

\emph{(3).} Como variĂĄveis jĂĄ pertencem Ă  linguagem, o argumento de (1) funciona igualmente dentro de $T$.

\smallskip
\textbf{Passo indutivo: $t=f(t_1,\ldots,t_m)$.}
Sejam $\bar x$ as variĂĄveis que ocorrem em $t$ (no sentido da sua convenĂ§ĂŁo: todas as variĂĄveis livres estĂŁo dentre $\bar x$).
Pela MetadefiniĂ§ĂŁo~\ref{mdef:formulaDoValorDeUmTermo},
\[
\delta_t(\bar x,y)\ \equiv\ \exists y_1\ldots\exists y_m\Bigl(\bigwedge_{i=1}^m \delta_{t_i}(\bar x,y_i)\ \wedge\ \Def_f(y_1,\ldots,y_m,y)\Bigr).
\]

\smallskip
\emph{(1) Trabalhando em $T'$.}
Pela hipĂłtese de induĂ§ĂŁo aplicada a cada $t_i$, temos
\[
T'\vdash \forall \bar x\,\forall y_i\bigl(t_i=y_i\leftrightarrow \delta_{t_i}(\bar x,y_i)\bigr)\qquad (i=1,\ldots,m).
\]
Usando essas equivalĂŞncias dentro do existencial (raciocĂ­nio proposicional), obtemos em $T'$:
\[
\delta_t(\bar x,y)\ \leftrightarrow\
\exists \bar y\Bigl(\bigwedge_{i=1}^m (t_i=y_i)\ \wedge\ \Def_f(\bar y,y)\Bigr).
\]
Agora, em $T'$ temos sempre
\[
\forall \bar u\,\forall v\bigl(f(\bar u)=v\leftrightarrow \Def_f(\bar u,v)\bigr),
\]
pois ou $f\in\mathcal L$ e entĂŁo isto ĂŠ tautolĂłgico pela convenĂ§ĂŁo $\Def_f(\bar u,v)\equiv f(\bar u)=v$,
ou $f$ ĂŠ novo e esta ĂŠ precisamente a sentenĂ§a definidora de $f$.
Logo podemos substituir $\Def_f(\bar y,y)$ por $f(\bar y)=y$ dentro de $T'$.

Com isso, e usando a congruĂŞncia para funĂ§Ăľes e a transitividade da igualdade,
temos a equivalĂŞncia (em $T'$):
\[
f(t_1,\ldots,t_m)=y\ \leftrightarrow\ \exists \bar y\Bigl(\bigwedge_{i=1}^m (t_i=y_i)\wedge f(\bar y)=y\Bigr).
\]
Juntando com a equivalĂŞncia anterior para $\delta_t(\bar x,y)$, concluĂ­mos
\[
T'\vdash \forall \bar x\,\forall y\bigl(t=y\leftrightarrow \delta_t(\bar x,y)\bigr),
\]
isto ĂŠ, (1).

\smallskip
\emph{(2) Trabalhando em $T$.}
Pela hipĂłtese de induĂ§ĂŁo, para cada $i\le m$,
\[
T\vdash \forall\bar x\,\exists!y_i\,\delta_{t_i}(\bar x,y_i).
\]
Fixe $\bar x$. Pela parte existencial, existem (Ăşnicos) $y_1,\ldots,y_m$ tais que $\delta_{t_i}(\bar x,y_i)$.
Agora mostramos que existe um Ăşnico $y$ tal que $\Def_f(y_1,\ldots,y_m,y)$:
\begin{itemize}
    \item Se $f$ ĂŠ novo, isto ĂŠ exatamente a condiĂ§ĂŁo da MetadefiniĂ§ĂŁo~\ref{mdef:definicaoExplicita}:
    $T\vdash \forall \bar u\,\exists!v\,\Def_f(\bar u,v)$.
    \item Se $f\in\mathcal L$, entĂŁo $\Def_f(\bar u,v)$ ĂŠ $f(\bar u)=v$.
    Para a existĂŞncia, a reflexividade da igualdade dĂĄ $T\vdash f(\bar u)=f(\bar u)$,
    e entĂŁo FO15, aplicado Ă  fĂłrmula $\phi(v)\equiv (f(\bar u)=v)$ com o termo $t=f(\bar u)$, dĂĄ $\exists v\,f(\bar u)=v$.
    Para a unicidade, de $f(\bar u)=v$ e $f(\bar u)=w$ obtemos $v=w$ por simetria e transitividade da igualdade.
\end{itemize}
Portanto, existe um Ăşnico $y$ tal que $\Def_f(y_1,\ldots,y_m,y)$.
Logo existe um Ăşnico $y$ tal que
\[
\exists y_1\ldots\exists y_m\Bigl(\bigwedge_{i=1}^m \delta_{t_i}(\bar x,y_i)\ \wedge\ \Def_f(y_1,\ldots,y_m,y)\Bigr),
\]
isto ĂŠ, $\exists!y\,\delta_t(\bar x,y)$.
Como $\bar x$ foi arbitrĂĄrio, concluĂ­mos (2):
\[
T\vdash \forall \bar x\,\exists!y\,\delta_t(\bar x,y).
\]

\smallskip
\emph{(3) Caso adicional: $t$ ĂŠ um $\mathcal L$-termo.}
Se $t$ ĂŠ um $\mathcal L$-termo, entĂŁo $f\in\mathcal L$ e cada $t_i$ tambĂŠm ĂŠ $\mathcal L$-termo.
Pela hipĂłtese de induĂ§ĂŁo (3) aplicada a cada $t_i$, temos as equivalĂŞncias $t_i=y_i\leftrightarrow \delta_{t_i}(\bar x,y_i)$ agora em $T$.
Repetindo o argumento de (1) inteiramente dentro de $T$ (e usando que $\Def_f(\bar y,y)$ ĂŠ $f(\bar y)=y$ por convenĂ§ĂŁo),
obtemos
\[
T\vdash \forall \bar x\,\forall y\bigl(t=y\leftrightarrow \delta_t(\bar x,y)\bigr),
\]
isto ĂŠ, (3).

Isso conclui a induĂ§ĂŁo e a prova do metalema.
\end{proof}

\subsection{EliminaĂ§ĂŁo de sĂ­mbolos definidos: fĂłrmulas}\label{subsec:eliminacaoSimbolosDefinidosFormulas}
\index{eliminaĂ§ĂŁo de sĂ­mbolos definidos}

Agora eliminamos sĂ­mbolos definidos de fĂłrmulas.

\begin{metadefinition}[TraduĂ§ĂŁo $\psi\mapsto \psi^\ast$]\label{mdef:traducaoFormulas}\index{eliminaĂ§ĂŁo de sĂ­mbolos definidos!fĂłrmulas}
    Seja $\psi$ uma fĂłrmula da linguagem $\mathcal L'$ cujas variĂĄveis livres estejam entre $x_1,\ldots,x_n$.
    Definimos por recursĂŁo uma $\mathcal L$-fĂłrmula $\psi^\ast$ (com as mesmas variĂĄveis livres) como segue.
    Em todos os casos atĂ´micos abaixo, as variĂĄveis auxiliares $u,v,u_1,\ldots,u_m$ sĂŁo escolhidas novas e distintas das variĂĄveis livres listadas.

    \begin{enumerate}[label=(\roman*)]
        \item Se $\psi$ ĂŠ $t_1=t_2$, entĂŁo
        \[
            \psi^\ast\ :=\ \exists u\,\exists v\bigl(\delta_{t_1}(\bar x,u)\wedge \delta_{t_2}(\bar x,v)\wedge u=v\bigr).
        \]

        \item Se $\psi$ ĂŠ um predicado $0$-ĂĄrio $R\in\mathcal L$, entĂŁo $\psi^\ast:=R$.

        \item Se $\psi$ ĂŠ $R(t_1,\ldots,t_m)$, em que $R\in\mathcal L$ ĂŠ predicado $m$-ĂĄrio e $m>0$, entĂŁo
        \[
            \psi^\ast\ :=\ \exists u_1\ldots \exists u_m\Bigl(\bigwedge_{i=1}^m \delta_{t_i}(\bar x,u_i)\ \wedge\ R(u_1,\ldots,u_m)\Bigr).
        \]

        \item Se $\psi$ ĂŠ um predicado $0$-ĂĄrio $P\in\mathcal L'\setminus\mathcal L$ definido por $\Def_P$, entĂŁo $\psi^\ast:=\Def_P$.

        \item Se $\psi$ ĂŠ $P(t_1,\ldots,t_m)$, em que $P\in\mathcal L'\setminus\mathcal L$ ĂŠ predicado $m$-ĂĄrio definido por $\Def_P$ e $m>0$, entĂŁo
        \[
            \psi^\ast\ :=\ \exists u_1\ldots \exists u_m\Bigl(\bigwedge_{i=1}^m \delta_{t_i}(\bar x,u_i)\ \wedge\ \Def_P(u_1,\ldots,u_m)\Bigr).
        \]

        \item Conectivos e quantificadores:
        \[
            (\neg\theta)^\ast:=\neg(\theta^\ast),\quad
            (\theta\square\eta)^\ast:=\theta^\ast\square\eta^\ast,
        \]
        para $\square\in\{\wedge,\vee,\rightarrow,\leftrightarrow\}$, e
        \[
            (\forall z\,\theta)^\ast:=\forall z\,(\theta^\ast),\qquad
            (\exists z\,\theta)^\ast:=\exists z\,(\theta^\ast).
        \]
    \end{enumerate}
\end{metadefinition}

\begin{metalemma}[EliminaĂ§ĂŁo de fĂłrmulas]\label{mlem:eliminacaoDeFormulas}\index{eliminaĂ§ĂŁo de sĂ­mbolos definidos!fĂłrmulas}
    Seja $T'$ uma extensĂŁo por definiĂ§ĂŁo de $T$.
    Para toda fĂłrmula $\psi(x_1, \dots, x_n)$ da linguagem $\mathcal L'$,
    existe uma fĂłrmula $\psi^\ast$ na linguagem $\mathcal L$ (como na MetadefiniĂ§ĂŁo~\ref{mdef:traducaoFormulas})
    com as mesmas variĂĄveis livres que $\psi$ tal que
    \[
        T'\vdash \forall x_1\ldots \forall x_n\,(\psi\leftrightarrow \psi^\ast).
    \]

    AlĂŠm disso, se $\psi$ ĂŠ uma fĂłrmula da linguagem $\mathcal L$, entĂŁo:
    \[
        T\vdash \forall x_1\ldots \forall x_n\,(\psi\leftrightarrow \psi^\ast).
    \]
\end{metalemma}

\begin{proof}
    Provaremos por induĂ§ĂŁo sobre a estrutura de $\psi$ a seguinte afirmaĂ§ĂŁo simultĂ˘nea.

    \smallskip
    \textbf{AfirmaĂ§ĂŁo A.} Para toda fĂłrmula $\psi(\bar x)$ de $\mathcal L'$, vale
    \[
        T'\vdash \forall \bar x\,(\psi\leftrightarrow \psi^\ast).
    \]

    \textbf{AfirmaĂ§ĂŁo B.} Se, alĂŠm disso, $\psi(\bar x)$ ĂŠ uma fĂłrmula de $\mathcal L$, entĂŁo
    \[
        T\vdash \forall \bar x\,(\psi\leftrightarrow \psi^\ast).
    \]

    \smallskip
    Em cada caso, $\psi^\ast$ ĂŠ a traduĂ§ĂŁo definida na MetadefiniĂ§ĂŁo~\ref{mdef:traducaoFormulas}.

    \smallskip
    \textbf{Caso atĂ´mico 1:} $\psi$ ĂŠ $t_1=t_2$.

    \emph{Prova de A.}
    Pelo item (1) do Metalema~\ref{mlem:eliminacaoDeTermos},
    \[
        T'\vdash \forall \bar x\,\forall u\,(t_1=u\leftrightarrow \delta_{t_1}(\bar x,u))
        \quad\text{e}\quad
        T'\vdash \forall \bar x\,\forall v\,(t_2=v\leftrightarrow \delta_{t_2}(\bar x,v)).
    \]
    Usando raciocĂ­nio proposicional (instĂ˘ncias de tautologias) e substitutividade da igualdade, obtemos em $T'$:
    \[
        T'\vdash \forall \bar x\Bigl(t_1=t_2 \ \leftrightarrow\ \exists u\,\exists v\bigl(\delta_{t_1}(\bar x,u)\wedge \delta_{t_2}(\bar x,v)\wedge u=v\bigr)\Bigr),
    \]
    e o lado direito ĂŠ, por definiĂ§ĂŁo, $\psi^\ast$.

    \emph{Prova de B.}
    Suponha agora que $\psi$ ĂŠ uma $\mathcal L$-fĂłrmula; entĂŁo $t_1$ e $t_2$ sĂŁo $\mathcal L$-termos.
    Pelo item (3) do Metalema~\ref{mlem:eliminacaoDeTermos},
    \[
        T\vdash \forall \bar x\,\forall u\,(t_1=u\leftrightarrow \delta_{t_1}(\bar x,u))
        \quad\text{e}\quad
        T\vdash \forall \bar x\,\forall v\,(t_2=v\leftrightarrow \delta_{t_2}(\bar x,v)).
    \]
    O mesmo raciocĂ­nio proposicional usado acima (agora dentro de $T$) dĂĄ
    \[
        T\vdash \forall \bar x\,(\psi\leftrightarrow \psi^\ast).
    \]

    \smallskip
    \textbf{Caso atĂ´mico 2:} $\psi$ ĂŠ uma fĂłrmula atĂ´mica com sĂ­mbolo de predicado $R\in\mathcal L$.

    Se $R$ tem aridade $0$, entĂŁo $\psi^\ast$ ĂŠ a prĂłpria fĂłrmula $R$, e A e B seguem por tautologia e generalizaĂ§ĂŁo.
    Suponha, portanto, que $\psi$ ĂŠ $R(t_1,\ldots,t_m)$ com $m>0$.

    \emph{Prova de A.}
    Pelo item (1) do Metalema~\ref{mlem:eliminacaoDeTermos}, para cada $i\le m$,
    \[
        T'\vdash \forall \bar x\,\forall u_i\,(t_i=u_i\leftrightarrow \delta_{t_i}(\bar x,u_i)).
    \]
    Usando essas equivalĂŞncias e a congruĂŞncia para predicados,
    obtemos em $T'$:
    \[
        T'\vdash \forall \bar x\Bigl(R(t_1,\ldots,t_m)\ \leftrightarrow\
        \exists u_1\ldots \exists u_m\bigl(\bigwedge_{i=1}^m \delta_{t_i}(\bar x,u_i)\wedge R(u_1,\ldots,u_m)\bigr)\Bigr),
    \]
    e o lado direito ĂŠ exatamente $\psi^\ast$.

    \emph{Prova de B.}
    Se $\psi$ ĂŠ uma $\mathcal L$-fĂłrmula, entĂŁo cada $t_i$ ĂŠ um $\mathcal L$-termo.
    Pelo item (3) do Metalema~\ref{mlem:eliminacaoDeTermos}, para cada $i$,
    \[
        T\vdash \forall \bar x\,\forall u_i\,(t_i=u_i\leftrightarrow \delta_{t_i}(\bar x,u_i)).
    \]
    Repetindo o mesmo argumento (congruĂŞncia para $R$ e raciocĂ­nio proposicional), agora em $T$,
    concluĂ­mos
    \[
        T\vdash \forall \bar x\,(\psi\leftrightarrow \psi^\ast).
    \]

    \smallskip
    \textbf{Caso atĂ´mico 3:} $\psi$ ĂŠ uma fĂłrmula atĂ´mica com sĂ­mbolo de predicado $P\in\mathcal L'\setminus\mathcal L$.

    Aqui somente A ĂŠ relevante (B nĂŁo se aplica, pois $\psi$ nĂŁo estĂĄ em $\mathcal L$).
    Se $P$ tem aridade $0$, entĂŁo $\psi$ ĂŠ $P$, $\psi^\ast$ ĂŠ $\Def_P$, e a conclusĂŁo segue do axioma definidor $P\leftrightarrow\Def_P$ e de generalizaĂ§ĂŁo.
    Suponha, portanto, que $\psi$ ĂŠ $P(t_1,\ldots,t_m)$ com $m>0$.

    Pelo axioma definidor de $P$ em $T'$, temos
    \[
        T'\vdash \forall \bar u\,(P(\bar u)\leftrightarrow \Def_P(\bar u)).
    \]
    Combinando isso com o item (1) do Metalema~\ref{mlem:eliminacaoDeTermos} aplicado a cada $t_i$
    e com raciocĂ­nio proposicional, obtemos
    \[
        T'\vdash \forall \bar x\Bigl(P(t_1,\ldots,t_m)\leftrightarrow
        \exists u_1\ldots \exists u_m\bigl(\bigwedge_{i=1}^m \delta_{t_i}(\bar x,u_i)\wedge \Def_P(u_1,\ldots,u_m)\bigr)\Bigr),
    \]
    que ĂŠ precisamente $\forall\bar x(\psi\leftrightarrow \psi^\ast)$.

    \smallskip
    \textbf{Conectivos.}
    Suponha que A (e, quando aplicĂĄvel, B) valem para $\theta$ e $\eta$.

    Para a negaĂ§ĂŁo, da hipĂłtese indutiva $T'\vdash \forall\bar x(\theta\leftrightarrow \theta^\ast)$
    e de uma instĂ˘ncia de tautologia proposicional
    $(A\leftrightarrow B)\rightarrow(\neg A\leftrightarrow \neg B)$,
    obtemos $T'\vdash \forall\bar x(\neg\theta\leftrightarrow \neg\theta^\ast)$,
    e pela definiĂ§ĂŁo $(\neg\theta)^\ast=\neg(\theta^\ast)$, isto ĂŠ exatamente A para $\neg\theta$.
    O mesmo argumento vale em $T$ quando $\theta$ estĂĄ em $\mathcal L$, provando B.

    Para os conectivos binĂĄrios $\square\in\{\wedge,\vee,\rightarrow,\leftrightarrow\}$,
    usamos instĂ˘ncias de tautologias do tipo
    \[
        (A\leftrightarrow A')\wedge(B\leftrightarrow B')\ \rightarrow\ ((A\square B)\leftrightarrow(A'\square B')),
    \]
    junto com as hipĂłteses indutivas para $\theta$ e $\eta$,
    e a definiĂ§ĂŁo $(\theta\square\eta)^\ast=\theta^\ast\square\eta^\ast$.
    Isso prova A (e tambĂŠm B quando $\theta,\eta$ sĂŁo fĂłrmulas de $\mathcal L$).

    \smallskip
    \textbf{Quantificadores.}
    Suponha que A vale para $\theta$.
    No caso de $\forall z\,\theta$, aplicamos a hipĂłtese indutiva a $\theta$ considerando suas variĂĄveis livres entre $\bar x,z$.
    Assim,
    \[
        T'\vdash \forall \bar x\,\forall z\,(\theta\leftrightarrow \theta^\ast).
    \]
    Pelas regras derivadas para quantificadores e para bicondicional, obtemos
    \[
        T'\vdash \forall \bar x\,(\forall z\,\theta \leftrightarrow \forall z\,\theta^\ast).
    \]
    Como $(\forall z\,\theta)^\ast=\forall z\,\theta^\ast$, isso prova A para $\forall z\,\theta$.

    O caso de $\exists z\,\theta$ ĂŠ anĂĄlogo, usando a equivalĂŞncia
    \[
        \forall z\,(\theta\leftrightarrow\theta^\ast)
    \]
    para obter
    \[
        \exists z\,\theta\leftrightarrow \exists z\,\theta^\ast.
    \]

    O mesmo argumento, realizado em $T$, prova B quando a fĂłrmula quantificada pertence Ă  linguagem $\mathcal L$.

    \smallskip
    Isso conclui a induĂ§ĂŁo e prova simultaneamente as duas afirmaĂ§Ăľes, logo o metalema.
\end{proof}

\subsection{ExtensĂľes por definiĂ§ĂŁo sĂŁo conservativas}\label{subsec:extensoesPorDefinicaoSaoConservativas}

\begin{metadefinition}[ExtensĂŁo conservativa]\label{mdef:extensaoConservativa}\index{extensĂŁo!conservativa}
    Seja $T$ uma teoria de primeira ordem com lĂŠxico $\mathcal L$, e seja $T'$ uma teoria com lĂŠxico $\mathcal L'$ estendendo $\mathcal L$.
    Dizemos que $T'$ ĂŠ uma \emph{extensĂŁo conservativa} de $T$ se cada axioma de $T$ ĂŠ tambĂŠm um axioma de $T'$ e, para cada $\mathcal L$-sentenĂ§a $\phi$,
    \[
        T'\vdash \phi \ \Longrightarrow\ T\vdash \phi.
    \]
\end{metadefinition}

O teorema a seguir formaliza a ideia de que definiĂ§Ăľes sĂŁo apenas abreviaĂ§Ăľes.

\begin{metatheorem}\label{mthm:conservatividadeDaExtensaoPorDefinicao}\index{definiĂ§ĂŁo!extensĂŁo por definiĂ§ĂŁo!conservatividade}
    Seja $T$ uma teoria de primeira ordem e $T'$ uma extensĂŁo por definiĂ§ĂŁo de $T$.
    EntĂŁo $T'$ ĂŠ uma extensĂŁo conservativa de $T$.
\end{metatheorem}

\begin{proof}
    Seja $\phi$ uma $\mathcal L$-sentenĂ§a e suponha $T'\vdash \phi$.
    Considere uma demonstraĂ§ĂŁo formal de $\phi$ a partir de $T'$,
    \[
        \theta_0,\theta_1,\ldots,\theta_n,
    \]
    em que $\theta_n$ ĂŠ $\phi$.

    Para cada $i$, considere a fĂłrmula $\theta_i^\ast$ obtida pela traduĂ§ĂŁo da MetadefiniĂ§ĂŁo~\ref{mdef:traducaoFormulas}.
    Cada $\theta_i^\ast$ ĂŠ uma fĂłrmula da linguagem $\mathcal L$.

    \smallskip
    \textbf{AfirmaĂ§ĂŁo.} Para todo $i\le n$, vale $T\vdash \theta_i^\ast$.

    \smallskip
    Provaremos por induĂ§ĂŁo em $i$.
    Suponha provado para todos $j<i$.

    \begin{itemize}
        \item Se $\theta_i$ ĂŠ um axioma de $T$, entĂŁo $\theta_i$ ĂŠ uma $\mathcal L$-sentenĂ§a.
        Pelo Metalema~\ref{mlem:eliminacaoDeFormulas} (aplicado a uma fĂłrmula jĂĄ na linguagem $\mathcal L$),
        temos $T\vdash \theta_i\leftrightarrow \theta_i^\ast$.
        Em particular, $\theta_i^\ast$ ĂŠ consequĂŞncia lĂłgica proposicional de $\theta_i$ e de uma instĂ˘ncia de tautologia;
        como $\theta_i$ ĂŠ axioma de $T$, concluĂ­mos $T\vdash \theta_i^\ast$.

        \item Se $\theta_i$ ĂŠ uma sentenĂ§a definidora (axioma de definiĂ§ĂŁo) de algum sĂ­mbolo novo, entĂŁo,
        $T\vdash \theta_i^\ast$.
        De fato, a traduĂ§ĂŁo substitui o sĂ­mbolo definido pela sua fĂłrmula definidora e substitui os termos variĂĄveis por suas fĂłrmulas de valor, que sĂŁo equivalentes a igualdades triviais pelo Metalema~\ref{mlem:eliminacaoDeTermos}.
        Como as fĂłrmulas definidoras jĂĄ sĂŁo fĂłrmulas de $\mathcal L$, a segunda parte do Metalema~\ref{mlem:eliminacaoDeFormulas} identifica cada fĂłrmula definidora com sua prĂłpria traduĂ§ĂŁo dentro de $T$.
        Assim, no caso de predicados, a traduĂ§ĂŁo da sentenĂ§a definidora se reduz em $T$ a uma instĂ˘ncia de $\alpha\leftrightarrow\alpha$; nos casos de funĂ§Ăľes e constantes, reduz-se do mesmo modo apĂłs eliminar o termo definido pelo lado que usa $\Def_f$ ou $\Def_c$.

        \item Se $\theta_i$ ĂŠ um axioma lĂłgico (no sentido do sistema fixado anteriormente),
        entĂŁo $T\vdash \theta_i^\ast$.
        A verificaĂ§ĂŁo ĂŠ sistemĂĄtica: usa-se que $\ast$ comuta com conectivos e quantificadores e que termos sĂŁo eliminados pelo Metalema~\ref{mlem:eliminacaoDeTermos}.
        Nos axiomas de congruĂŞncia, se o sĂ­mbolo envolvido ĂŠ novo, usa-se ainda que a fĂłrmula definidora ĂŠ uma $\mathcal L$-fĂłrmula, a Lei da igualdade de Leibniz e, no caso de sĂ­mbolos funcionais, a hipĂłtese de unicidade exigida na MetadefiniĂ§ĂŁo~\ref{mdef:definicaoExplicita}.
        Assim, $T\vdash \theta_i^\ast$.

        \item Se $\theta_i$ ĂŠ obtida por \emph{modus ponens} a partir de $\theta_j$ e $\theta_k=\theta_j\rightarrow \theta_i$,
        com $j,k<i$, entĂŁo por hipĂłtese de induĂ§ĂŁo $T\vdash \theta_j^\ast$ e $T\vdash (\theta_j\rightarrow \theta_i)^\ast$.
        Pela definiĂ§ĂŁo de $\ast$, $(\theta_j\rightarrow \theta_i)^\ast$ ĂŠ $\theta_j^\ast\rightarrow \theta_i^\ast$.
        Logo, por \emph{modus ponens}, $T\vdash \theta_i^\ast$.

        \item Se $\theta_i$ ĂŠ obtida pela regra de introduĂ§ĂŁo do quantificador universal, entĂŁo existem fĂłrmulas $\chi,\eta$, uma variĂĄvel $x$ e $j<i$ tais que
        $\theta_j$ ĂŠ $\chi\rightarrow\eta$, $\theta_i$ ĂŠ $\chi\rightarrow\forall x\,\eta$, e $x$ nĂŁo ocorre livre em $\chi$.
        Pela hipĂłtese de induĂ§ĂŁo,
        \[
            T\vdash (\chi\rightarrow\eta)^\ast,
        \]
        isto ĂŠ, $T\vdash \chi^\ast\rightarrow\eta^\ast$.
        Como $\chi^\ast$ tem as mesmas variĂĄveis livres que $\chi$, a variĂĄvel $x$ nĂŁo ocorre livre em $\chi^\ast$.
        Pela regra de introduĂ§ĂŁo do quantificador universal,
        \[
            T\vdash \chi^\ast\rightarrow\forall x\,\eta^\ast,
        \]
        que ĂŠ exatamente $\theta_i^\ast$.

        \item Se $\theta_i$ ĂŠ obtida pela regra de introduĂ§ĂŁo do quantificador existencial, entĂŁo existem fĂłrmulas $\chi,\eta$, uma variĂĄvel $x$ e $j<i$ tais que
        $\theta_j$ ĂŠ $\chi\rightarrow\eta$, $\theta_i$ ĂŠ $\exists x\,\chi\rightarrow\eta$, e $x$ nĂŁo ocorre livre em $\eta$.
        Pela hipĂłtese de induĂ§ĂŁo, $T\vdash \chi^\ast\rightarrow\eta^\ast$.
        Como $\eta^\ast$ tem as mesmas variĂĄveis livres que $\eta$, a variĂĄvel $x$ nĂŁo ocorre livre em $\eta^\ast$.
        Pela regra de introduĂ§ĂŁo do quantificador existencial,
        \[
            T\vdash \exists x\,\chi^\ast\rightarrow\eta^\ast,
        \]
        que ĂŠ exatamente $\theta_i^\ast$.
    \end{itemize}

    Isso conclui a induĂ§ĂŁo, e em particular $T\vdash \theta_n^\ast=\phi^\ast$.

    \smallskip
    Resta observar que, como $\phi$ jĂĄ estĂĄ na linguagem $\mathcal L$ e ĂŠ sentenĂ§a,
    o Metalema~\ref{mlem:eliminacaoDeFormulas} dĂĄ $T\vdash \phi\leftrightarrow \phi^\ast$,
    e portanto $\phi$ ĂŠ consequĂŞncia proposicional de $\phi^\ast$.
    Como $T\vdash \phi^\ast$, segue $T\vdash \phi$.
\end{proof}
