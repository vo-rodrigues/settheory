\section{Relações de ordem}
Nesta seção, introduziremos as noções básicas de pré-ordem, ordem parcial e ordem total, bem como suas versões estritas e conceitos associados.

\begin{definition}[Pré-ordem]\index{pré-ordem}
    Uma \emph{pré-ordem} é um par $(P, \leq)$ onde $P$ é um conjunto e $\leq$ é uma relação binária em $P$ tal que para todo $x, y, z \in P$:
    \begin{itemize}
        \item $x \leq x$ (reflexividade)
        \item Se $x \leq y$ e $y \leq z$, então $x \leq z$ (transitividade).
    \end{itemize}
\end{definition}

\begin{definition}[Ordem parcial]\index{ordem parcial}
    Uma \emph{ordem parcial} é um par $(P, \leq)$ onde $P$ é um conjunto e $\leq$ é uma pré-ordem e, além disso, satisfaz a seguinte condição.
    \begin{itemize}
        \item Se $x \leq y$ e $y \leq x$, então $x = y$ (antissimetria).
    \end{itemize}
\end{definition}

\begin{definition}[Ordem total]\index{ordem total}\index{ordem linear}
    Uma \emph{ordem total}, ou \emph{ordem linear}, é um par $(P, \leq)$ onde $P$ é um conjunto e $\leq$ é uma ordem parcial e, além disso, satisfaz a seguinte condição.
    \begin{itemize}
        \item Para todo $x, y \in P$, ou $x \leq y$ ou $y \leq x$ (totalidade).
    \end{itemize}
\end{definition}

Há também versões estritas.

\begin{definition}\index{ordem parcial estrita}
    Uma \emph{ordem parcial estrita} é um par $(P, <)$ onde $P$ é um conjunto e $<$ é uma relação binária em $P$ tal que para todo $x, y, z \in P$, valem as seguintes condições.
    \begin{itemize}
        \item Não existe $x$ tal que $x < x$ (irreflexividade).
        \item Se $x < y$ e $y < z$, então $x < z$ (transitividade).
    \end{itemize}
\end{definition}

\begin{definition}\index{ordem total estrita}
    Uma \emph{ordem total estrita} é um par $(P, <)$ onde $P$ é um conjunto e $<$ é uma ordem parcial estrita e, além disso, satisfaz a seguinte condição.
    \begin{itemize}
        \item Para todo $x, y \in P$, ou $x < y$ ou $y < x$ ou $x = y$ (totalidade).
    \end{itemize}
\end{definition}

As versões estritas e não estritas estão em correspondência natural.

\begin{definition}[Ordem parcial estrita induzida]
    Seja $(P, \leq)$ uma ordem parcial.
    A ordem parcial estrita associada a $(P, \leq)$ é o par $(P, <)$ em que $<$ é a relação binária em $P$ definida por $x < y$ se e somente se $x \leq y$ e $x \neq y$.
\end{definition}

\begin{definition}[Ordem parcial não estrita induzida]
    Seja $(P, <)$ uma ordem parcial estrita.
    A ordem parcial associada a $(P, <)$ é o par $(P, \leq)$ em que $\leq$ é a relação binária em $P$ definida por $x \leq y$ se e somente se $x < y$ ou $x = y$.
\end{definition}

As seguintes duas proposições são deixadas como exercício ao leitor (Exercício~\ref{ex:ordens-duais}).

\begin{proposition}\label{prop:ordens-duais}
    Seja $(P, \leq)$ uma ordem parcial.
    \begin{enumerate}[label=(\alph*)]
        \item A ordem parcial estrita induzida por $(P, \leq)$ é de fato uma ordem parcial estrita.
        \item Seja $(P, <)$ a ordem parcial estrita induzida por $(P, \leq)$.
        Então a ordem parcial não estrita induzida por $(P, <)$ é $(P, \leq)$.
        \item $(P, \leq)$ é uma ordem total se e somente se $(P, <)$ é uma ordem total estrita.
    \end{enumerate}
\end{proposition}

Dualmente, temos a seguinte proposição.

\begin{proposition}\label{prop:ordens-duais-2}
    Seja $(P, <)$ uma ordem parcial estrita.
    \begin{enumerate}[label=(\alph*)]
        \item A ordem parcial induzida por $(P, <)$ é de fato uma ordem parcial.
        \item Seja $(P, \leq)$ a ordem parcial induzida por $(P, <)$.
        Então a ordem parcial estrita induzida por $(P, \leq)$ é $(P, <)$.
        \item $(P, <)$ é uma ordem total estrita se e somente se $(P, \leq)$ é uma ordem total.
    \end{enumerate}
\end{proposition}

Por isso, ao se estudar ordens parciais, não há perda em se restringir ao estudo de ordens estritas ou não estritas.
Além disso, ao longo deste livro, ao estabelecer que $\leq$ é uma ordem parcial, iremos, sem aviso, utilizar a ordem estrita $<$ associada a $\leq$ e vice-versa.
Outros símbolos que poderão ser usados para denotar ordens parciais fazendo jus a esta convenção são $\preceq$ e $\prec$, $\sqsubseteq$ e $\sqsubset$, dentre outros.

\begin{example}
    Seja $A$ qualquer conjunto.
    Define-se $\subseteq_A=\{(X, Y) \in \mathcal{P}(A) \times \mathcal{P}(A) : X \subseteq Y\}$.
    Então $(\mathcal{P}(A), \subseteq_A)$ é uma ordem parcial.
    
    É usual denotá-la por $(\mathcal{P}(A), \subseteq)$, omitindo o subíndice $A$.
\end{example}


Abaixo, definiremos os conceitos de cota superior, cota inferior, máximo, mínimo, elemento maximal, elemento minimal, supremo e ínfimo de um conjunto em uma pré-ordem.
Para ordens parciais estritas, tais conceitos são definidos a partir da ordem parcial não estrita associada.

\begin{definition}\index{cota superior}\index{cota inferior}\index{máximo}\index{mínimo}\index{elemento maximal}\index{elemento minimal}\index{supremo}\index{ínfimo}
    Seja $(P, \leq)$ uma pré-ordem e $A \subseteq P$. Dizemos que:
    \begin{itemize}
        \item $x$ é uma \emph{cota superior} de $A$ se para todo $a \in A$, $a \leq x$;
        \item $x$ é uma \emph{cota inferior} de $A$ se para todo $a \in A$, $x \leq a$;
        \item $x$ é um \emph{máximo} de $A$, ou um \emph{maior elemento} de $A$, se é uma cota superior de $A$ e $x \in A$;
        \item $x$ é um \emph{mínimo} de $A$, ou um \emph{menor elemento} de $A$, se é uma cota inferior de $A$ e $x \in A$;
        \item $x$ é \emph{maximal} em $A$ se $x \in A$ e para todo $a \in A$, tal que $x \leq a$, segue que $a\leq x$;
        \item $x$ é \emph{minimal} em $A$ se $x \in A$ e para todo $a \in A$, tal que $a \leq x$, segue que $x \leq a$;
        \item $x$ é um \emph{supremo} de $A$ se é uma menor cota superior de $A$, ou seja, $x$ é uma cota superior de $A$ e para toda cota superior $y$ de $A$, segue que $x \leq y$;
        \item $x$ é um \emph{ínfimo} de $A$ se é uma maior cota inferior de $A$, ou seja, $x$ é uma cota inferior de $A$ e para toda cota inferior $y$ de $A$, segue que $y \leq x$.
    \end{itemize}
\end{definition}

Deixamos também os seguintes fatos básicos para verificação do leitor (Exercício~\ref{ex:ordens-basicas}).

\begin{proposition}\label{prop:ordens-basicas}
    Seja $(P, \leq)$ uma ordem parcial e $A \subseteq P$.
    \begin{itemize}
        \item Se existe um máximo de $A$, então ele é o único máximo, e o denotamos por $\max A$.
        \item Se existe um mínimo de $A$, então ele é o único mínimo, e o denotamos por $\min A$.
        \item Se existe um supremo de $A$, então ele é o único supremo, e o denotamos por $\sup A$.
        \item Se existe um ínfimo de $A$, então ele é o único ínfimo, e o denotamos por $\inf A$.
        \item Se $x \in A$, então $x$ é um elemento maximal de $A$ se e somente se para todo $a \in A$, se $x \leq a$, então $x=a$.
        \item Se $x \in A$, então $x$ é um elemento minimal de $A$ se e somente se para todo $a \in A$, se $a \leq x$, então $x=a$.
    \end{itemize}
\end{proposition}

Finalmente, introduzimos a noção de isomorfismo.
\begin{definition}[Isomorfismo entre relações binárias]\index{isomorfismo de ordens}
    Sejam $P$ e $Q$ conjuntos e $R$ e $S$ relações binárias em $P$ e $Q$, respectivamente.
    Um \emph{isomorfismo} entre $(P, R)$ e $(Q, S)$ é uma bijeção $f : P \to Q$ tal que para todo $x, y \in P$, $x R y$ se e somente se $f(x) S f(y)$.
\end{definition}

Para ordens totais, basta a verificação de um dos lados da implicação, como mostra a seguinte proposição.

\begin{proposition}
    Sejam $P$ e $Q$ conjuntos e $\leq$, $\preceq$ ordens totais em $P$ e $Q$, respectivamente.
    Uma bijeção $f : P \to Q$ é um isomorfismo entre $(P, \leq)$ e $(Q, \preceq)$ se e somente se para todo $x, y \in P$, $x \leq y$ implica $f(x) \preceq f(y)$.
\end{proposition}

\begin{proof}
    Suponha que $f(x) \preceq f(y)$.
    Se $y<x$, teríamos $f(y) \preceq f(x)$. Porém, como $y<x$ e $f$ é injetora, segue que $f(y) \prec f(x)$, o que é uma contradição.
\end{proof}
\subsection*{Exercícios}
\begin{exercise}\label{ex:ordens-duais}
    Prove as proposições \ref{prop:ordens-duais} e \ref{prop:ordens-duais-2}.
\end{exercise}
\begin{exercise}\label{ex:ordens-basicas}
    Prove a Proposição~\ref{prop:ordens-basicas}.
\end{exercise}
\begin{exercise}
    Seja $(P, \leq)$ uma ordem total, $A\subseteq P$ e $x \in P$.
    Prove que $x$ é elemento maximal de $A$ se e somente se $x$ é o máximo de $A$.
    Enuncie e prove um resultado análogo para elementos minimais e mínimos.
\end{exercise}

\begin{exercise}
    Seja $(P, \leq)$ uma ordem total.
    Mostre que para todo $x, y \in P$, não vale simultaneamente $x<y$ e $y\leq x$
\end{exercise}

\begin{exercise}
    Seja $(P, <)$ uma ordem parcial estrita tal que qualquer conjunto de $2$ elementos de $P$ tem um elemento mínimo.
    Prove que $(P, <)$ é uma ordem total estrita.
\end{exercise}

\begin{exercise}
    Mostre que se $A$ tem dois elementos distintos, então $(\mathcal{P}(A), \subseteq)$ não é uma ordem total.
\end{exercise}

\begin{exercise}
    Seja $P$ um conjunto e $Q \subseteq P$.
    Seja $R$ uma relação binária em $P$.
    Prove as seguintes afirmações.

    \begin{itemize}
        \item Se $(P, R)$ é uma pré-ordem, então $(Q, R \cap (Q \times Q))$ é uma pré-ordem.
        \item Se $(P, R)$ é uma ordem parcial, então $(Q, R \cap (Q \times Q))$ é uma ordem parcial.
        \item Se $(P, R)$ é uma ordem total, então $(Q, R \cap (Q \times Q))$ é uma ordem total.
        \item Se $(P, R)$ é uma ordem parcial estrita, então $(Q, R \cap (Q \times Q))$ é uma ordem parcial estrita.
        \item Se $(P, R)$ é uma ordem total estrita, então $(Q, R \cap (Q \times Q))$ é uma ordem total estrita.
    \end{itemize}
\end{exercise}
