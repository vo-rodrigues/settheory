\section{Conjuntos enumerĂĄveis}
\index{conjunto enumerĂĄvel}

\begin{definition}\index{enumerabilidade}
    Um conjunto $A$ ĂŠ \emph{enumerĂĄvel} se ĂŠ finito, ou se ĂŠ equipotente a $\omega$.
\end{definition}

\begin{lemma}\label{lem:subconjunto-enumeravel}
    Todo subconjunto de $\omega$ ĂŠ enumerĂĄvel.
\end{lemma}
\begin{proof}
    Seja $X\subseteq \omega$.
    Se $X$ ĂŠ finito, entĂŁo $X$ ĂŠ enumerĂĄvel.
    Caso contrĂĄrio, $X$ ĂŠ infinito.

    Recursivamente, defina $f:\omega\to X$ tal que, para todo $n \in \omega$,

    \begin{align*}
        f(n) = \begin{cases}
            \min(X\setminus \{f(m) : m<n\}), & \text{se } X\setminus \{f(m) : m<n\}\neq \emptyset\\
            \min X, & \text{caso contrĂĄrio}
        \end{cases}
    \end{align*}

    Por induĂ§ĂŁo em $n$, vĂŞ-se que $f|n$ ĂŠ injetora e que $f(n)=\min(X\setminus \{f(m) : m<n\})$.
    Isso ĂŠ imediato para $n=0$.
    
    Sendo verdade para $n$, temos que $f(n)>f(m)$ para todo $m<n$, de modo que $f|(n+1)$ ĂŠ injetora.
    Como $X$ ĂŠ infinito, $X\setminus \{f(m) : m<n+1\}\neq \emptyset$, ou $f|(n+1): n+1\to X$ seria uma bijeĂ§ĂŁo.
    Assim, $f(n+1)=\min(X\setminus \{f(m) : m<n+1\})$.

    Portanto, temos que $f$ ĂŠ uma funĂ§ĂŁo injetora de $\omega$ em $X$.
    Deste modo, $\omega\preceq X$.
    Como $X\subseteq \omega$, segue do Teorema de Cantor-SchrĂśder-Bernstein que $X\approx \omega$, e, portanto, $X$ ĂŠ enumerĂĄvel.
\end{proof}

\begin{proposition}
Seja $A$ um conjunto.
SĂŁo equivalentes:
\begin{enumerate}[label=(\alph*)]
    \item $A$ ĂŠ enumerĂĄvel.
    \item Existe $f:A\rightarrow \omega$ uma funĂ§ĂŁo injetora.
\end{enumerate}
\end{proposition}
\begin{proof}
    Basta provar que (b) implica (a).
    Seja $f:A\to \omega$ uma funĂ§ĂŁo injetora e $X=f[A]$.
    Como $X\subseteq \omega$, pelo Lema~\ref{lem:subconjunto-enumeravel}, $X$ ĂŠ enumerĂĄvel.
    Assim, $X$ ĂŠ finito ou $X\approx \omega$.
    Em qualquer caso, como $A\approx X$, segue a tese.
\end{proof}

\begin{proposition}
    $\omega\times \omega$ ĂŠ infinito enumerĂĄvel.
\end{proposition}
\begin{proof}
    Note que $f:\omega\times \omega\to \omega$ dada por $f(m,n)=2^m(2n+1)-1$ ĂŠ uma funĂ§ĂŁo bijetora.

    Para a injetividade, sejam $m, m', n, n'\in \omega$ tais que $f(m,n)=f(m',n')$.
    Suponha por absurdo que $m<m'$.
    EntĂŁo:
    \begin{align*}
        &2^m(2n+1)-1 = 2^{m'}(2n'+1)-1 \geq 2^{m+1}-1\\
        \implies &2n+1 =2^{m'-m}(2n'+1).
    \end{align*}
    um absurdo, pois o resto da divisĂŁo por $2$ de $2n+1$ ĂŠ $1$, enquanto o resto da divisĂŁo por $2$ de $2^{m'-m}(2n'+1)$ ĂŠ $0$.
    Analogamente, se $m'>m$,temos um absurdo.

    Assim, $m=m'$. Cancelando $2^m=2^{m'}$, temos que $2n+1=2n'+1$, e, portanto, $n=n'$.

    Para a sobrejetividade, lembre-se que dizemos que um nĂşmero natural $a$ divide um nĂşmero natural $b$ se existe um nĂşmero natural $c$ tal que $ac=b$.
    Denotamos ``$a$ divide $b$'' por $a|b$.
    Note que, se $a|b$ e $b\neq 0$, temos que existe $c$ tal que $b=ac\geq a1=a$.
    
    Dado $l\in \omega$, seja $X=\{m\in \omega : 2^m|(l+1)\}$.
    Temos que, para todo $m \in \omega$, $2^m>m$, logo, $X\subseteq l+1$.
    AlĂŠm disso, $X\neq \emptyset$, pois $2^0=1$ divide $l+1$.
    Logo, $X$ possui um mĂĄximo, $m$.

    Seja $x\in \omega$ tal que $l+1=2^mx$.
    Existe $r<2$ e $n\in \omega$ tais que $x=2n+r$.
    Se $r=0$, entĂŁo $l+1=2^m2n=2^{m+1}n$, o que ĂŠ impossĂ­vel, pois $m$ ĂŠ o mĂĄximo de $X$.
    Assim, $r=1$, e, portanto, $l+1=2^m(2n+1)$.
    Logo, $f(m,n)=l$.
\end{proof}
\subsection*{ExercĂ­cios}

\begin{exercise}
    Mostre que a funĂ§ĂŁo $f$ construĂ­da na prova do Lema~\ref{lem:subconjunto-enumeravel} no caso em que $X$ ĂŠ infinito ĂŠ uma funĂ§ĂŁo bijetora de $\omega$ em $X$.
    AlĂŠm disso, determine uma funĂ§ĂŁo $G:X^{<\omega}\to X$ tal que para todo $n \in \omega$, $f(n)=G(f|n)$.
\end{exercise}
\begin{exercise}
    Prove que se $A$ ĂŠ um conjunto para o qual existe $f:\omega\rightarrow A$ uma funĂ§ĂŁo sobrejetora, entĂŁo $A$ ĂŠ enumerĂĄvel.
\end{exercise}
\begin{exercise}
    Prove que se $A, B$ sĂŁo enumerĂĄveis, entĂŁo $A\cup B$ ĂŠ enumerĂĄvel.
\end{exercise}
\begin{exercise}
    Suponha que $(A_n)_{n\in\omega}$ seja uma sequĂŞncia de conjuntos enumerĂĄveis para a qual existe uma famĂ­lia $(f_n)_{n\in\omega}$ tal que para cada $n\in\omega$, $f_n:A_n\rightarrow \omega$ ĂŠ uma funĂ§ĂŁo injetora.
    Prove que $\bigcup_{n\in\omega} A_n$ ĂŠ enumerĂĄvel.

    ObservaĂ§ĂŁo: nĂŁo ĂŠ possĂ­vel demonstrar, em ZF, que se $(B_n)_{n\in\omega}$ ĂŠ uma sequĂŞncia de conjuntos enumerĂĄveis, entĂŁo $\bigcup_{n\in\omega} B_n$ ĂŠ enumerĂĄvel.
    Na presenĂ§a do Axioma da Escolha, tal enunciado ĂŠ verdadeiro.
\end{exercise}
\begin{exercise}
    Prove que se $A, B$ sĂŁo enumerĂĄveis, entĂŁo $A\times B$ ĂŠ enumerĂĄvel.
\end{exercise}
\begin{exercise}
    Seja $(A_n)_{n\in\omega}$ uma sequĂŞncia de conjuntos enumerĂĄveis.
    Prove que para todo $k\in \omega$, $\prod_{n<k} A_n$ ĂŠ enumerĂĄvel.
    
    Em particular, conclua que se $A$ ĂŠ enumerĂĄvel, entĂŁo $A^k$ ĂŠ enumerĂĄvel para todo $k\in \omega$.
\end{exercise}

\begin{exercise}
    Seja $A$ um conjunto enumerĂĄvel.
    Mostre que $A^{<\omega}=\bigcup_{n\in\omega} A^n$ ĂŠ enumerĂĄvel.

    O seguinte roteiro ĂŠ sugerido.
    
    \begin{enumerate}
        \item Mostre que se $f:A^n\rightarrow \omega$ ĂŠ uma funĂ§ĂŁo, entĂŁo $f \in \mathcal P(A^{<\omega}\times \omega)$.
        Conclua que existe o conjunto $B=\bigcup_{n\in\omega}\omega^{A^n}$.

        \item Fixe $h:A\to \omega$ uma funĂ§ĂŁo injetora e $g:\omega\times\omega\to \omega$ uma funĂ§ĂŁo injetora.

        Defina $\phi:B\to B$ da seguinte forma: para todo $n\in \omega$ e toda funĂ§ĂŁo $u:A^n\to \omega$,
        a funĂ§ĂŁo $\phi(u):A^{n+1}\to \omega$ ĂŠ dada por
        \[
            \phi(u)(s)=g(u(s|n),h(s(n))).
        \]

        Observe que $\phi$ estĂĄ bem definida e que, se $u$ ĂŠ uma funĂ§ĂŁo injetora, entĂŁo $\phi(u)$ ĂŠ uma funĂ§ĂŁo injetora.

        \item Fixe uma funĂ§ĂŁo injetora $f_0:A^0\to \omega$.
        Aplique o ExercĂ­cio~\ref{ex:recursionComposition} Ă  funĂ§ĂŁo $\phi:B\to B$ e defina, para cada $n\in\omega$,
        \[
            f_n=\phi^n(f_0).
        \]
        Conclua que, para cada $n\in\omega$, $f_n:A^n\to \omega$ ĂŠ uma funĂ§ĂŁo injetora.
        \item Conclua que $A^{<\omega}$ ĂŠ enumerĂĄvel.
    \end{enumerate}
\end{exercise}

\begin{exercise}
    Seja $A$ um conjunto enumerĂĄvel.
    Mostre que o conjunto $[A]^{<\omega}=\bigcup_{n\in\omega}[A]^n=\{s \in \mathcal P(A) : s \text{ ĂŠ finito}\}$ ĂŠ enumerĂĄvel.
\end{exercise}
