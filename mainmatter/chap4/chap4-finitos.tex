\section{Conjuntos finitos}\label{sec:conjuntosFinitos}

Pela DefiniĂ§ĂŁo~\ref{def:ordemDosNaturais} e pela ProposiĂ§ĂŁo~\ref{prop:naturaisSaoSubconjuntosDeOmega}, se $n>0$ Ă© um nĂșmero natural, entĂŁo $n=\{0,1,\ldots,n-1\}$, ou, mais formalmente, $n=\{m\in\omega : m<n\}$.

Dessa forma, podemos definir a noĂ§ĂŁo de conjunto finito.

\begin{definition}[$Z^--P$]\label{def:conjuntoFinito}\index{finitude}\index{infinitude}
    Dizemos que um conjunto $A$ Ă© \emph{finito} se existe um nĂșmero natural $n$ tal que $A\approx n$.
    Nesse caso, dizemos que a cardinalidade de $A$ Ă© $n$.
    Caso contrĂĄrio, dizemos que $A$ Ă© \emph{infinito}.
\end{definition}

Note que, para cada nĂșmero natural $n$, $n$ Ă© finito, pois $n\approx n$, e a cardinalidade de $n$ Ă© $n$.

Intuitivamente, se um conjunto tem 35 elementos, ele nĂŁo pode ter 33 elementos, nem 38 elementos.
Veremos que, se um conjunto Ă© finito, o natural $n$ com o qual ele Ă© equipotente Ă© Ășnico.

Primeiro provaremos o seguinte lema.

\begin{lemma}[$Z^--P$]\label{lem:subconjuntoProprioDeNaturalNaoEquipotente}
    Para todo $n \in \omega$ e para todo $X\subseteq n$, se $X\neq n$, entĂŁo $X\not\approx n$.
\end{lemma}
\begin{proof}
    Seguimos por induĂ§ĂŁo em $n$.
    Para $n=0$, o Ășnico subconjunto de $n$ Ă© $n$, assim, o resultado Ă© trivial.

    Suponha que o resultado seja verdadeiro para todo $m<n$.
    Seja $X\subseteq n+1=n\cup\{n\}$ tal que $X\neq n+1$.
    Suponha por absurdo que existe uma funĂ§ĂŁo bijetora $f: n+1\to X$.

    Caso 1: $n\notin X$.
    Nesse caso, $f(n)\neq n$.
    Como $f$ Ă© injetora, $X'=f[n]$ Ă© um subconjunto de $n$.
    AlĂ©m disso, a restriĂ§ĂŁo $f\restriction n:n\to X'$ Ă© bijetora, de modo que $X'\approx n$.
    Como $f(n)\in n$ e $f(n)\notin X'$, temos $X'\neq n$, um absurdo.

    Caso 2: $n\in X$.
    Seja $k\in n+1$ tal que $f(k)=n$.
    Considere $g:n\to X\setminus \{n\}$ dada por
    \[
        g(m) = \begin{cases}
            f(m), & \text{se } m\neq k\\
            f(n), & \text{se } m=k.
        \end{cases}
    \]

    Da injetividade de $f$, decorre que $g$ Ă© injetora.

    AlĂ©m disso, $g$ Ă© sobrejetora sobre $X\setminus\{n\}$.
    De fato, seja $y\in X\setminus\{n\}$.
    Como $f$ Ă© sobrejetora, existe $m\in n+1$ tal que $f(m)=y$.
    Se $m<n$ e $m\neq k$, entĂŁo $g(m)=y$.
    Se $m=n$, entĂŁo $g(k)=f(n)=y$.

    Portanto, $g:n\to X\setminus\{n\}$ Ă© bijetora.
    Como $X\setminus\{n\}\subseteq n$, segue pela hipĂłtese indutiva que $X\setminus\{n\}=n$.
    Logo, $X=n\cup\{n\}=n+1$, um absurdo.
\end{proof}


\begin{corollary}[$Z^-$]Sobre finitude, temos o seguinte.\label{cor:finitoCardinal}
    \begin{enumerate}[label=(\alph*)]
        \item Para todos $n, m\in \omega$, $n\preceq m$ se, e somente se $n\leq m$.
        \item Para todos $n, m\in \omega$, $n\approx m$ se, e somente se $n=m$.
        \item Para todo conjunto finito $A$, existe um Ășnico nĂșmero natural $n$ tal que $A\approx n$.
        \item $\omega$ Ă© infinito.
    \end{enumerate}
\end{corollary}
\begin{proof}
    Para (a), seja $n, m\in \omega$.
    Se $m<n$, temos que $m \not \approx n$, pois $m$ Ă© um subconjunto prĂłprio de $n$.
    Como $m\preceq n$, segue que $n\not\preceq m$.

    Assim, se $n\preceq m$, entĂŁo $n\leq m$.
    Reciprocamente, se $n\leq m$, entĂŁo $n\subseteq m$, e, assim, $n\preceq m$.

    Para (b), seja $n, m\in \omega$.
    Temos que, do item (a):
    \[
        n\approx m \Longleftrightarrow n\preceq m \text{ e } m\preceq n \Longleftrightarrow n\leq m \text{ e } m\leq n \Longleftrightarrow n=m.
    \]

    Para (c), seja $A$ um conjunto finito.
    Por definiĂ§ĂŁo, existe um nĂșmero natural $n$ tal que $A\approx n$.
    Se $m, n$ sĂŁo nĂșmeros naturais tais que $A\approx m$ e $A\approx n$, entĂŁo $m\approx n$, pelo item (b), e, assim, $m=n$.

    Para (d), suponha por absurdo que $\omega$ Ă© finito.
    EntĂŁo, existe um nĂșmero natural $n$ tal que $\omega\approx n$.
    PorĂ©m, $n\subsetneq n+1\subseteq \omega$, e, assim, pelo Teorema de Cantor-SchrĂ¶der-Bernstein, $n+1\approx \omega\approx n$, um absurdo por (c). 
\end{proof}

\begin{lemma}[$Z^-$]\label{prop:finitoBoaOrdem}
    Seja $(A, <)$ uma ordem total estrita nĂŁo vazia tal que $A$ Ă© finito.
    EntĂŁo $A$ possui um elemento mĂ­nimo e um elemento mĂĄximo.
    Em particular, toda ordem total estrita finita Ă© uma boa ordem.
\end{lemma}
\begin{proof}
    Provaremos por induĂ§ĂŁo na cardinalidade de $A$.
    Se $A\approx 0$, Ă© trivial.

    Suponha que a afirmaĂ§ĂŁo seja verdadeira para toda ordem total estrita de cardinalidade $n$.
    Seja $A$ uma ordem total estrita de cardinalidade $n+1$.
    Tome $a\in A$ qualquer e seja $A' = A\setminus\{a\}$ com a ordem restrita.
    $A'$ Ă© uma ordem total estrita de cardinalidade $n$.
    Se $A'=\emptyset$, entĂŁo $A=\{a\}$, e $a$ Ă© o elemento mĂ­nimo e o elemento mĂĄximo de $A$.
    Caso contrĂĄrio, $A'$ possui um elemento mĂ­nimo $m$ e um elemento mĂĄximo $M$.
    
    Se $a<m$, entĂŁo $a$ Ă© o elemento mĂ­nimo de $A$.
    Caso contrĂĄrio, $m$ Ă© o elemento mĂ­nimo de $A$.

    Da mesma forma, se $a>M$, entĂŁo $a$ Ă© o elemento mĂĄximo de $A$.
    Caso contrĂĄrio, $M$ Ă© o elemento mĂĄximo de $A$.
\end{proof}

\begin{proposition}[$Z^-$]\label{prop:boaOrdemFinitaIsomorfaANatural}
    Seja $(A, <)$ uma ordem total estrita tal que $A$ Ă© finito.
    EntĂŁo existe um Ășnico nĂșmero natural $n$ tal que $(A, <)$ Ă© isomorfo Ă  $(n, <)$.
\end{proposition}
\begin{proof}
A unicidade segue da unicidade do nĂșmero natural equipotente a $A$.

Para a existĂȘncia, utilizaremos induĂ§ĂŁo na cardinalidade de $A$.

Se $A\approx 0$, temos que $A=\emptyset$, logo, $(A, <)=(0, <)$.

Suponha que o resultado seja verdadeiro para todo conjunto finito de cardinalidade $n$.
Seja $A$ uma ordem total de $n+1$ elementos.
Seja $a=\max A$.
Seja $A' = A\setminus\{a\}$ com a ordem restrita.
Por hipĂłtese indutiva, existe $\phi': A'\to n$ um isomorfismo de ordens.
Defina $f': A\to n+1$ dada por
\[
    f'(x) = \begin{cases}
        \phi'(x), & \text{se } x\neq a\\
        n, & \text{se } x=a.
    \end{cases}
\]
Verifica-se facilmente que $f'$ Ă© um isomorfismo de ordens de $(A, <)$ em $(n+1, <)$.
\end{proof}
